3. SqlReaderクラス (src/sql_reader.py)

```python
"""
SqlReaderクラス
SnowflakeのステージからSQLファイルを読み込む共通クラス

【説明】
- Snowflakeの内部ステージに配置されたSQLファイルを直接読み込む
- ファイルをローカルにダウンロードせず、ステージから直接内容を取得
- 名前付きクエリをサポート（ファイル内の特定のクエリを取得）
"""

from typing import Dict, Optional
from snowflake.snowpark import Session


class SqlReader:
    """
    SnowflakeステージからSQL文を読み込むクラス
    """

    def __init__(self, session: Session, stage_name: str):
        """
        コンストラクタ

        Args:
            session: Snowparkセッション
            stage_name: ステージ名（例: '@MY_STAGE'）
        """
        self.session = session
        self.stage_name = stage_name
        self._sql_cache: Dict[str, str] = {}  # クエリキャッシュ

    def read_sql_file(self, file_name: str, query_name: Optional[str] = None) -> str:
        """
        ステージからSQLファイルを読み込み、指定されたクエリを返す

        Args:
            file_name: 読み込むSQLファイル名（例: 'queries.sql'）
            query_name: 取得するクエリの名前（コメントで指定）
                        Noneの場合はファイル全体を返す

        Returns:
            str: SQL文

        Raises:
            FileNotFoundError: ファイルが存在しない場合
            ValueError: 指定されたクエリ名が見つからない場合

        【使用例】
            reader = SqlReader(session, '@MY_STAGE')
            # 特定のクエリを取得
            sql = reader.read_sql_file('test_queries.sql', 'query_get_all_employees')
            # ファイル全体を取得
            sql_all = reader.read_sql_file('test_queries.sql')
        """
        cache_key = f"{file_name}_{query_name if query_name else 'full'}"

        # キャッシュにあればそれを返す
        if cache_key in self._sql_cache:
            return self._sql_cache[cache_key]

        # ステージ上のファイル存在確認
        if not self._file_exists(file_name):
            raise FileNotFoundError(
                f"ファイル '{file_name}' がステージ '{self.stage_name}' に見つかりません"
            )

        # ファイルの内容を読み込む（ステージから直接読み込み）
        file_content = self._read_file_from_stage(file_name)

        if query_name:
            # 特定のクエリを抽出
            sql = self._extract_query(file_content, query_name)
            if not sql:
                raise ValueError(
                    f"クエリ名 '{query_name}' がファイル '{file_name}' に見つかりません"
                )
        else:
            # ファイル全体を返す
            sql = file_content

        # キャッシュに保存
        self._sql_cache[cache_key] = sql
        return sql

    def _file_exists(self, file_name: str) -> bool:
        """
        ステージ上にファイルが存在するか確認

        Args:
            file_name: ファイル名

        Returns:
            bool: 存在する場合はTrue
        """
        try:
            # LIST コマンドでファイル一覧を取得
            result = self.session.sql(
                f"LIST {self.stage_name}/{file_name}"
            ).collect()
            return len(result) > 0
        except Exception:
            return False

    def _read_file_from_stage(self, file_name: str) -> str:
        """
        ステージからファイルを直接読み込む（ローカルダウンロードなし）

        Args:
            file_name: ファイル名

        Returns:
            str: ファイル内容
        """
        # Snowflakeのファイル読み込み関数を使用
        # METADATA$FILE_ROW_NUMBER を使用して行番号を取得
        query = f"""
        SELECT 
            METADATA$FILE_ROW_NUMBER as row_num,
            $1 as line
        FROM {self.stage_name}/{file_name}
        ORDER BY row_num
        """

        df = self.session.sql(query)
        rows = df.collect()

        # 行を結合して完全なファイル内容を復元
        lines = [row["LINE"] for row in rows]
        return "\n".join(lines)

    def _extract_query(self, file_content: str, query_name: str) -> Optional[str]:
        """
        ファイル内容から指定された名前のクエリを抽出

        Args:
            file_content: ファイル内容
            query_name: 抽出するクエリ名

        Returns:
            Optional[str]: 抽出されたSQL、見つからない場合はNone

        抽出ルール:
        - コメント行 '-- query_name' の次の行から次のコメント行までを抽出
        - セミコロン(;)で終わる
        """
        lines = file_content.split('\n')
        result_lines = []
        in_target_query = False

        for line in lines:
            stripped = line.strip()

            # クエリ名の開始を検出
            if stripped.startswith('-- ') and stripped[3:].strip() == query_name:
                in_target_query = True
                continue

            # 別のクエリ名が始まったら終了
            if in_target_query and stripped.startswith('-- '):
                break

            # 対象クエリ内の行を収集
            if in_target_query:
                result_lines.append(line)

        if not result_lines:
            return None

        return '\n'.join(result_lines).strip()
```

4. SqlRunnerクラス (src/sql_runner.py)

```python
"""
SqlRunnerクラス
SQL実行を共通化するクラス

【説明】
- CRUD操作の共通メソッドを提供
- トランザクション管理機能
- バルクインサート、バルクアップデート対応
- エラーハンドリングとロールバック機能
"""

from typing import Any, Dict, List, Optional, Union
from snowflake.snowpark import Session
from snowflake.snowpark.table import Table
import pandas as pd


class SqlRunner:
    """
    SQL実行共通クラス
    """

    def __init__(self, session: Session, auto_commit: bool = True):
        """
        コンストラクタ

        Args:
            session: Snowparkセッション
            auto_commit: 自動コミットフラグ（True: 各操作後にコミット、False: 手動コミット必要）
        """
        self.session = session
        self.auto_commit = auto_commit
        self._transaction_active = False

    def execute_query(self, sql: str, params: Optional[Dict[str, Any]] = None) -> List[Dict[str, Any]]:
        """
        SQLクエリを実行し、結果を返す（SELECT系）

        Args:
            sql: 実行するSQL
            params: バインドパラメータ（名前付きパラメータ）

        Returns:
            List[Dict[str, Any]]: クエリ結果

        【使用例】
            sql = "SELECT * FROM employees WHERE department = :dept"
            result = runner.execute_query(sql, {"dept": "IT"})
        """
        try:
            if params:
                # 名前付きパラメータをバインド
                df = self.session.sql(sql, params=params)
            else:
                df = self.session.sql(sql)

            result = df.collect()
            # Rowオブジェクトを辞書に変換
            return [row.as_dict() for row in result]

        except Exception as e:
            print(f"クエリ実行エラー: {e}")
            raise

    def execute_update(
        self, 
        sql: str, 
        params: Optional[Union[Dict[str, Any], List[Dict[str, Any]]]] = None
    ) -> int:
        """
        UPDATE/INSERT/DELETEを実行し、影響を受けた行数を返す

        Args:
            sql: 実行するSQL
            params: バインドパラメータ
                    - 単一操作: Dict形式
                    - バルク操作: List[Dict]形式

        Returns:
            int: 影響を受けた行数

        【使用例】
            # 単一更新
            sql = "UPDATE employees SET salary = :salary WHERE id = :id"
            count = runner.execute_update(sql, {"salary": 50000, "id": 1})

            # バルク更新
            params_list = [
                {"salary": 50000, "id": 1},
                {"salary": 60000, "id": 2}
            ]
            total_count = runner.execute_update(sql, params_list)
        """
        try:
            if isinstance(params, list) and len(params) > 0:
                # バルク操作
                total_count = 0
                for single_params in params:
                    count = self._execute_single_update(sql, single_params)
                    total_count += count
                return total_count
            else:
                # 単一操作
                return self._execute_single_update(sql, params)

        except Exception as e:
            print(f"更新実行エラー: {e}")
            if not self.auto_commit:
                self.rollback()
            raise

    def _execute_single_update(self, sql: str, params: Optional[Dict[str, Any]] = None) -> int:
        """
        単一の更新SQLを実行

        Args:
            sql: SQL
            params: パラメータ

        Returns:
            int: 影響行数
        """
        if params:
            df = self.session.sql(sql, params=params)
        else:
            df = self.session.sql(sql)

        # SQL_RETURNINGで影響行数を取得
        result = df.collect()
        
        # 自動コミット設定
        if self.auto_commit and not self._transaction_active:
            self.commit()

        # 影響行数を返す（Snowflakeの場合は件数が直接返ることが多い）
        return len(result) if result else 0

    # ===== CRUD操作 =====

    def insert(
        self, 
        table_name: str, 
        data: Union[Dict[str, Any], List[Dict[str, Any]]]
    ) -> int:
        """
        テーブルにデータを挿入

        Args:
            table_name: テーブル名
            data: 挿入データ
                  - 単一レコード: Dict形式
                  - 複数レコード: List[Dict]形式

        Returns:
            int: 挿入された行数
        """
        if isinstance(data, list):
            # 複数レコードの挿入
            return self._bulk_insert(table_name, data)
        else:
            # 単一レコードの挿入
            return self._single_insert(table_name, data)

    def _single_insert(self, table_name: str, data: Dict[str, Any]) -> int:
        """
        単一レコード挿入
        """
        columns = list(data.keys())
        values = list(data.values())

        # プレースホルダー作成
        placeholders = [f":{col}" for col in columns]

        sql = f"""
        INSERT INTO {table_name} 
        ({', '.join(columns)}) 
        VALUES ({', '.join(placeholders)})
        """

        return self.execute_update(sql, data)

    def _bulk_insert(self, table_name: str, data_list: List[Dict[str, Any]]) -> int:
        """
        バルクインサート（複数レコード一括挿入）
        """
        if not data_list:
            return 0

        # DataFrameを作成して一括挿入
        df = pd.DataFrame(data_list)
        
        # Snowpark DataFrameに変換
        snowpark_df = self.session.create_dataframe(df)
        
        # テーブルに書き込み
        snowpark_df.write.mode("append").save_as_table(table_name)
        
        # 自動コミット
        if self.auto_commit and not self._transaction_active:
            self.commit()

        return len(data_list)

    def update(
        self,
        table_name: str,
        data: Dict[str, Any],
        condition: str,
        condition_params: Optional[Dict[str, Any]] = None
    ) -> int:
        """
        テーブルデータを更新

        Args:
            table_name: テーブル名
            data: 更新するデータ（カラム名と値のDict）
            condition: WHERE条件（例: "id = :id"）
            condition_params: 条件のパラメータ

        Returns:
            int: 更新された行数

        【使用例】
            data = {"salary": 60000}
            condition = "id = :emp_id AND department = :dept"
            params = {"emp_id": 1, "dept": "IT"}
            count = runner.update("employees", data, condition, params)
        """
        # SET句の作成
        set_clause = ", ".join([f"{col} = :{col}" for col in data.keys()])

        sql = f"""
        UPDATE {table_name}
        SET {set_clause}
        WHERE {condition}
        """

        # パラメータを結合
        params = {**data}
        if condition_params:
            params.update(condition_params)

        return self.execute_update(sql, params)

    def delete(
        self,
        table_name: str,
        condition: str,
        condition_params: Optional[Dict[str, Any]] = None
    ) -> int:
        """
        テーブルデータを削除

        Args:
            table_name: テーブル名
            condition: WHERE条件
            condition_params: 条件のパラメータ

        Returns:
            int: 削除された行数

        【使用例】
            condition = "id = :id AND department = :dept"
            params = {"id": 1, "dept": "IT"}
            count = runner.delete("employees", condition, params)
        """
        sql = f"DELETE FROM {table_name} WHERE {condition}"
        return self.execute_update(sql, condition_params)

    def select(
        self,
        table_name: str,
        columns: List[str] = None,
        condition: Optional[str] = None,
        condition_params: Optional[Dict[str, Any]] = None,
        limit: Optional[int] = None
    ) -> List[Dict[str, Any]]:
        """
        テーブルからデータを選択

        Args:
            table_name: テーブル名
            columns: 取得するカラム（Noneの場合は全カラム）
            condition: WHERE条件
            condition_params: 条件のパラメータ
            limit: 取得件数制限

        Returns:
            List[Dict[str, Any]]: 取得データ

        【使用例】
            result = runner.select(
                "employees",
                columns=["id", "name", "salary"],
                condition="department = :dept AND salary > :min_salary",
                condition_params={"dept": "IT", "min_salary": 50000},
                limit=10
            )
        """
        # カラム指定
        col_str = ", ".join(columns) if columns else "*"

        sql = f"SELECT {col_str} FROM {table_name}"

        if condition:
            sql += f" WHERE {condition}"

        if limit:
            sql += f" LIMIT {limit}"

        return self.execute_query(sql, condition_params)

    # ===== トランザクション管理 =====

    def begin_transaction(self) -> None:
        """
        トランザクション開始
        """
        self.session.sql("BEGIN").collect()
        self._transaction_active = True
        print("トランザクション開始")

    def commit(self) -> None:
        """
        トランザクションコミット
        """
        self.session.sql("COMMIT").collect()
        self._transaction_active = False
        print("コミット完了")

    def rollback(self) -> None:
        """
        トランザクションロールバック
        """
        self.session.sql("ROLLBACK").collect()
        self._transaction_active = False
        print("ロールバック完了")

    # ===== ユーティリティメソッド =====

    def execute_sql_file(
        self,
        sql_reader: SqlReader,
        file_name: str,
        query_name: str,
        params: Optional[Dict[str, Any]] = None
    ) -> List[Dict[str, Any]]:
        """
        SQLファイルからクエリを読み込んで実行

        Args:
            sql_reader: SqlReaderインスタンス
            file_name: SQLファイル名
            query_name: 実行するクエリ名
            params: パラメータ

        Returns:
            List[Dict[str, Any]]: 実行結果
        """
        # SQLを読み込み
        sql = sql_reader.read_sql_file(file_name, query_name)
        
        # 実行
        return self.execute_query(sql, params)

    def table_exists(self, table_name: str) -> bool:
        """
        テーブル存在確認

        Args:
            table_name: テーブル名

        Returns:
            bool: 存在する場合はTrue
        """
        try:
            self.session.table(table_name).limit(1).collect()
            return True
        except:
            return False
```

5. ユーティリティ (src/utils/snowpark_utils.py)

```python
"""
Snowparkユーティリティ
セッション管理などの共通機能
"""

import toml
from snowflake.snowpark import Session
from typing import Optional, Dict, Any


def create_session_from_toml(toml_path: str = "../config/secrets.toml", profile: str = "default") -> Session:
    """
    TOMLファイルからSnowflakeセッションを作成

    Args:
        toml_path: TOMLファイルのパス
        profile: プロファイル名

    Returns:
        Session: Snowparkセッション

    【説明】
    - secrets.tomlから接続情報を読み込んでセッションを作成
    - 本番環境では環境変数やシークレットサービスからの取得を推奨
    """
    with open(toml_path, 'r') as f:
        config = toml.load(f)

    connection_params = config.get(profile, {})
    
    # 必須パラメータの確認
    required_params = ['account', 'user', 'password']
    missing_params = [p for p in required_params if p not in connection_params]
    
    if missing_params:
        raise ValueError(f"必須パラメータが不足: {missing_params}")

    return Session.builder.configs(connection_params).create()


def close_session(session: Session) -> None:
    """
    セッションをクローズ

    Args:
        session: Snowparkセッション
    """
    if session:
        session.close()
        print("セッションをクローズしました")
```

            "name": "トランザクションテスト",
            "department": "Test",

        }

        print(f"トランザクション内での確認: {result}")
        
        print("\n--- ロールバック実行 ---")
        runner.rollback()
        
        # 確認（ロールバック後は見えない）

        print(f"ロールバック後の確認（空のはず）: {result}")

        runner.begin_transaction()
        runner.insert("employees", {"id": 21, "name": "コミットテスト", "department": "Test", "salary": 88888})
        runner.commit()
        
        # 確認（コミット後は見える）

# Snowpark共通モジュール

## 概要
SnowflakeのSnowparkを使用した共通処理モジュールです。
ステージからのSQLファイル読み込み、CRUD操作、トランザクション管理を提供します。

## ファイル構成
- `sql_reader.py`: ステージからSQLを読み込むクラス
- `sql_runner.py`: SQL実行共通クラス（CRUD、トランザクション）
- `utils/snowpark_utils.py`: セッション管理ユーティリティ

## セットアップ手順

### 1. 設定ファイルの編集
`config/secrets.toml` にSnowflake接続情報を設定

### 2. テスト環境の準備
```sql
-- Snowflake上で実行
CREATE DATABASE IF NOT EXISTS TEST_DB;
USE TEST_DB.PUBLIC;

-- テストテーブル作成
CREATE OR REPLACE TABLE employees (
    id INTEGER PRIMARY KEY,
    name VARCHAR(100),
    department VARCHAR(50),
    salary NUMBER(10, 2),
    created_at TIMESTAMP_NTZ DEFAULT CURRENT_TIMESTAMP()
);

-- ステージ作成
CREATE OR REPLACE STAGE sql_stage
    DIRECTORY = (ENABLE = TRUE);
```

3. SQLファイルのアップロード

```bash
# SnowSQLを使用
PUT file://sql/test_queries.sql @sql_stage;
```

4. テスト実行

```bash
cd tests
python test_snowpark_common.py
```

主要クラスの説明

SqlReader

· ステージからSQLファイルを直接読み込み
· 名前付きクエリの抽出機能
· キャッシュ機能でパフォーマンス最適化

SqlRunner

· CRUD操作の共通メソッド
· バルクインサート/アップデート対応
· トランザクション管理（BEGIN/COMMIT/ROLLBACK）
· パラメータバインディング対応

注意事項

· 本番環境ではsecrets.tomlの代わりに環境変数やシークレット管理サービスの使用を推奨
· ステージ上のSQLファイルは適切な権限設定を行うこと
· 大量データのバルク操作はパフォーマンスに注意

この共通モジュールは以下の特徴を持っています：

1. **SqlReader**: ステージから直接SQLを読み込む（ローカルダウンロードなし）

3. **トランザクション管理**: BEGIN/COMMIT/ROLLBACKを完全サポート
4. **バルク操作**: 複数レコードの一括挿入に対応
5. **テスト完全**: 各機能のテストケースを提供
