四、日志类

common/logger.py

import datetime
from snowflake.snowpark.files import SnowflakeFile


class StageLogger:
    """
    Snowflake Stage にログを書き込む共通クラス

    目的
    ----------
    ・共通クラス(SqlReader / SqlRunner)の入口と出口ログ
    ・エラー発生時のログ
    ・運用時のトラブルシュート

    特徴
    ----------
    SnowflakeFile.open() を使用して
    直接 Stage に書き込みます
    """

    def __init__(self, stage_path="@MAIN_H.GOLD.SQL_STAGE/logs/app.log"):
        self.stage_path = stage_path

    def log(self, message: str):
        """
        ログ出力
        """

        timestamp = datetime.datetime.now().isoformat()

        log_line = f"[{timestamp}] {message}\n"

        with SnowflakeFile.open(self.stage_path, "a") as f:
            f.write(log_line)

import hashlib
from snowflake.snowpark.files import SnowflakeFile
from common.logger import StageLogger


class SqlReader:
    """
    Stage 上の SQL ファイルを読み込む共通クラス

    機能
    ----------
    1. Stage から SQL ファイルを直接読み込み
    2. SQL 名で管理
    3. MD5 による変更検知
    4. キャッシュ管理
    5. SQL Hot Reload
    """

    def __init__(self, stage_path: str):

        """
        stage_path 例
        @MAIN_H.GOLD.SQL_STAGE/sql/user_sql.sql
        """

        self.stage_path = stage_path
        self.cache = {}
        self.md5 = None
        self.logger = StageLogger()

    def _calc_md5(self, content: str):

        """
        SQLファイル内容の MD5 を計算
        """

        return hashlib.md5(content.encode()).hexdigest()

    def _load_sql_file(self):

        """
        Stage から SQL ファイルを読み込み
        """

        self.logger.log("SqlReader: load sql file start")

        with SnowflakeFile.open(self.stage_path, "r") as f:
            content = f.read()

        md5 = self._calc_md5(content)

        if md5 == self.md5:
            return

        self.md5 = md5

        sql_map = {}

        current_name = None
        buffer = []

        for line in content.splitlines():

            line = line.strip()

            if line.startswith("-- name:"):

                if current_name:
                    sql_map[current_name] = "\n".join(buffer)

                current_name = line.replace("-- name:", "").strip()

                buffer = []

            else:
                buffer.append(line)

        if current_name:
            sql_map[current_name] = "\n".join(buffer)

        self.cache = sql_map

        self.logger.log("SqlReader: sql cache updated")

    def get_sql(self, name: str):

        """
        SQL取得
        """

        self._load_sql_file()

        if name not in self.cache:
            raise Exception(f"SQL not found: {name}")

        return self.cache[name]

from snowflake.snowpark import Session
from common.logger import StageLogger


class SqlRunner:
    """
    SQL 実行共通クラス

    機能
    ----------
    SELECT
    INSERT
    UPDATE
    DELETE
    TRANSACTION
    """

    def __init__(self, session: Session):

        """
        Snowflake 内部実行の場合
        session は既存のものを使用できます
        """

        self.session = session
        self.logger = StageLogger()

    def query(self, sql: str, params: dict = None):

        """
        SELECT 専用
        """

        self.logger.log("SqlRunner query start")

        df = self.session.sql(sql, params)

        result = df.collect()

        self.logger.log("SqlRunner query end")

        return result

    def execute(self, sql: str, params: dict = None):

        """
        INSERT / UPDATE / DELETE
        """

        self.logger.log("SqlRunner execute start")

        result = self.session.sql(sql, params).collect()

        affected = 0

        if result:
            try:
                affected = result[0][0]
            except:
                affected = len(result)

        self.logger.log(f"SqlRunner execute end affected={affected}")

        return affected

    def transaction(self, func):

        """
        トランザクション制御
        """

        self.logger.log("transaction start")

        try:

            self.session.sql("BEGIN").collect()

            result = func()

            self.session.sql("COMMIT").collect()

            self.logger.log("transaction commit")

            return result

        except Exception as e:

            self.session.sql("ROLLBACK").collect()

            self.logger.log(f"transaction rollback {str(e)}")

            raise

七、Streamlit 测试 UI

Snowflake Streamlit

app/streamlit_app.py

import streamlit as st
from snowflake.snowpark.context import get_active_session

from common.sql_reader import SqlReader
from common.sql_runner import SqlRunner


session = get_active_session()

reader = SqlReader("@MAIN_H.GOLD.SQL_STAGE/sql/user_sql.sql")

runner = SqlRunner(session)


st.title("Snowpark SQL Framework Test")


menu = st.sidebar.selectbox(
    "menu",
    ["select_all", "select_by_id", "insert", "update", "delete", "transaction_test"]
)

if menu == "select_all":

    sql = reader.get_sql("select_all")

    result = runner.query(sql)

    st.write(result)


elif menu == "select_by_id":

    user_id = st.number_input("user id", 1)

    sql = reader.get_sql("select_by_id")

    result = runner.query(sql, {"user_id": user_id})

    st.write(result)


elif menu == "insert":

    user_id = st.number_input("user id")
    name = st.text_input("name")
    email = st.text_input("email")

    if st.button("insert"):

        sql = reader.get_sql("insert_user")

        count = runner.execute(
            sql,
            {
                "user_id": user_id,
                "user_name": name,
                "email": email,
            },
        )

        st.success(f"insert count {count}")


elif menu == "update":

    user_id = st.number_input("user id")
    name = st.text_input("name")
    email = st.text_input("email")

    if st.button("update"):

        sql = reader.get_sql("update_user")

        count = runner.execute(
            sql,
            {
                "user_id": user_id,
                "user_name": name,
                "email": email,
            },
        )

        st.success(f"update count {count}")


elif menu == "delete":

    user_id = st.number_input("user id")

    if st.button("delete"):

        sql = reader.get_sql("delete_user")

        count = runner.execute(sql, {"user_id": user_id})

        st.success(f"delete count {count}")


elif menu == "transaction_test":

    if st.button("run transaction"):

        def process():

            sql1 = reader.get_sql("insert_user")

            runner.execute(
                sql1,
                {
                    "user_id": 999,
                    "user_name": "tx_user",
                    "email": "tx@test.com",
                },
            )

            sql2 = reader.get_sql("update_user")

            runner.execute(
                sql2,
                {
                    "user_id": 1,
                    "user_name": "tx_update",
                    "email": "tx_update@test.com",
                },
            )

        runner.transaction(process)

        st.success("transaction success")
