### 3. 第二步：共通封装类 (`snowflake_utils.py`)

这是核心文件，包含了**连接单例、自动重连、显式事务和通用 CRUD**。

```python
import logging
from snowflake.snowpark import Session
from snowflake.snowpark.exceptions import SnowparkException
from snowflake.snowpark.functions import col
from contextlib import contextmanager

# 配置生产级日志
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')
logger = logging.getLogger("SnowparkCore")

class SnowparkManager:
    """
    Snowpark 生产级共通类
    支持：自动连接管理、CRUD 封装、显式事务控制、异常统一处理
    """
    def __init__(self, connection_params):
        self.params = connection_params
        self.session = None

    def get_session(self) -> Session:
        """获取或创建有效的 Session (支持自动重连)"""
        if self.session is None or self.session.is_closed():
            try:
                self.session = Session.builder.configs(self.params).create()
                logger.info("Snowflake Session 成功建立")
            except Exception as e:
                logger.error(f"无法建立 Snowflake 连接: {e}")
                raise
        return self.session

    @contextmanager
    def transaction_scope(self):
        """
        事务上下文管理器
        用法: with manager.transaction_scope() as session: ...
        自动处理 COMMIT 和 ROLLBACK
        """
        sess = self.get_session()
        try:
            sess.sql("BEGIN").collect()
            logger.info("事务已开启")
            yield sess
            sess.sql("COMMIT").collect()
            logger.info("事务已提交")
        except Exception as e:
            sess.sql("ROLLBACK").collect()
            logger.error(f"事务执行失败，已触发回滚: {e}")
            raise

    # --- CRUD 共通方法 ---

    def select(self, table_name: str, filters: dict = None, limit: int = None):
        """通用查询：返回 Snowpark DataFrame"""
        sess = self.get_session()
        try:
            df = sess.table(table_name)
            if filters:
                for k, v in filters.items():
                    df = df.filter(col(k) == v)
            if limit:
                df = df.limit(limit)
            return df
        except Exception as e:
            logger.error(f"查询表 {table_name} 失败: {e}")
            raise

    def insert_pandas(self, df, table_name: str, mode="append"):
        """将 Pandas DataFrame 写入 Snowflake"""
        sess = self.get_session()
        try:
            sess.write_pandas(df, table_name, auto_create_table=False, overwrite=(mode=="overwrite"))
            logger.info(f"成功向 {table_name} 写入数据")
        except Exception as e:
            logger.error(f"写入数据失败: {e}")
            raise

    def update_record(self, table_name: str, assignments: dict, condition: str):
        """通用更新"""
        sess = self.get_session()
        try:
            return sess.table(table_name).update(assignments, condition)
        except Exception as e:
            logger.error(f"更新失败: {e}")
            raise

    def close(self):
        """显式关闭连接"""
        if self.session:
            self.session.close()
            logger.info("Session 已关闭")

```

### 4. 第三步：测试与演示界面 (`app_test.py`)

利用 Streamlit 构建一个可视化的控制台，测试上述类的可靠性。

```python
import streamlit as st
import pandas as pd
from snowflake_utils import SnowparkManager

st.set_page_config(page_title="Snowpark 生产类测试", layout="wide")
st.title("🧪 Snowpark 共通类可靠性测试面板")

# --- 缓存连接对象 ---
@st.cache_resource
def init_manager():
    # 自动从 .streamlit/secrets.toml 读取
    return SnowparkManager(st.secrets["snowflake"])

manager = init_manager()

# --- 侧边栏状态 ---
with st.sidebar:
    st.header("系统状态")
    if st.button("检查连接"):
        sess = manager.get_session()
        st.success(f"连接成功! 当前库: {sess.get_current_database()}")

# --- 测试模块 1: 查询与缓存展示 ---
st.header("1. 数据读取测试 (Read)")
if st.button("从 Snowflake 加载最新库存"):
    with st.spinner("查询中..."):
        # 获取 Snowpark DF 并转为 Pandas 用于展示
        df_result = manager.select("PRODUCT_INVENTORY").to_pandas()
        st.session_state['data'] = df_result
        st.dataframe(df_result, use_container_width=True)

# --- 测试模块 2: 事务与异常回滚 (Transaction) ---
st.header("2. 事务可靠性测试 (Transaction)")
col1, col2 = st.columns(2)

with col1:
    st.subheader("场景 A: 模拟成功更新")
    new_price = st.number_input("设置 ID=1 的新价格", value=5500.0)
    if st.button("提交价格更新"):
        try:
            with manager.transaction_scope():
                manager.update_record("PRODUCT_INVENTORY", {"PRICE": new_price}, "ID = 1")
            st.success("更新成功并已提交！")
        except Exception as e:
            st.error(f"失败: {e}")

with col2:
    st.subheader("场景 B: 模拟失败回滚")
    st.warning("此操作会先更新 ID=2 的库存，随后抛出异常，测试数据是否回滚。")
    if st.button("触发强制失败"):
        try:
            with manager.transaction_scope() as sess:
                # 步骤 1: 更新数据
                manager.update_record("PRODUCT_INVENTORY", {"STOCK": 999}, "ID = 2")
                # 步骤 2: 抛出异常
                raise RuntimeError("人工模拟系统崩溃")
        except Exception as e:
            st.error(f"捕获到异常: {e}")
            st.info("💡 检查数据库，你会发现 ID=2 的库存没有变成 999，回滚生效！")

# --- 测试模块 3: 批量写入 (Create) ---
st.header("3. 批量写入测试 (Create)")
if st.button("新增随机产品"):
    new_id = int(pd.Timestamp.now().timestamp())
    test_df = pd.DataFrame([{
        "ID": new_id, "NAME": f"新产品_{new_id}", "STOCK": 50, "PRICE": 199.0
    }])
    manager.insert_pandas(test_df, "PRODUCT_INVENTORY")
    st.success("批量写入完成")

```

### 5. 如何部署运行？

1. **准备环境**:
```bash
pip install streamlit snowflake-snowpark-python pandas

```


2. **配置密码**:
在项目目录下创建 `.streamlit/secrets.toml`:
```toml
[snowflake]
account = "你的账号ID"
user = "用户名"
password = "密码"
role = "ACCOUNTADMIN"
warehouse = "COMPUTE_WH"
database = "PROD_DEMO_DB"
schema = "PUBLIC"

```


3. **启动测试界面**:
```bash
streamlit run app_test.py

```



### 生产级要点说明：

* **资源缓存**: 使用 `@st.cache_resource` 包装 `SnowparkManager`，确保整个 Web 应用生命周期内只维护一个连接池，极大提升响应速度。
* **上下文事务**: 使用 `with manager.transaction_scope()` 确保了业务逻辑的原子性，这对于金融或库存类应用至关重要。
* **解耦设计**: `snowflake_utils.py` 不依赖任何 Streamlit 组件，这意味着你可以直接在定时任务、FastAPI 接口或任何其他 Python 脚本中复用这个文件。
