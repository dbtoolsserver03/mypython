1. common/sql_reader.py（ステージからSQLを読む共通）

```python
# -*- coding: utf-8 -*-
"""
Snowflake上のステージから指定されたSQLファイルを読み込む共通クラス。
ファイルをローカルにダウンロードせず、内容を文字列として取得する。
"""

import snowflake.connector

class SqlReader:
    """
    ステージ上のSQLファイルを読み込むクラス。
    例: stage_path = '@MY_STAGE/sql/insert_test_data.sql'
    """

    @staticmethod
    def read_sql_from_stage(conn, stage_file_path: str) -> str:
        """
        ステージ上のSQLファイルを読み込み、内容を文字列で返す。
        :param conn: Snowflakeコネクションオブジェクト
        :param stage_file_path: '@ステージ名/パス/ファイル名.sql' の形式
        :return: SQL文の文字列
        """
        try:
            with conn.cursor() as cur:
                # ファイル内容を1行ずつ取得
                cur.execute(f"SELECT $1 FROM {stage_file_path}")
                rows = cur.fetchall()
                # 全行を結合（改行で連結）
                sql_content = "\n".join([row[0] for row in rows])
                return sql_content
        except Exception as e:
            raise RuntimeError(f"ステージからのSQL読込失敗: {stage_file_path} - {str(e)}")
```

2. common/sql_runner.py（SQL実行共通）

```python
# -*- coding: utf-8 -*-
"""
SQL実行共通クラス。
SELECT/INSERT/UPDATE/DELETE、バッチ処理、トランザクション制御を提供する。
"""

import snowflake.connector

class SqlRunner:
    """
    SnowflakeでのSQL実行を共通化するクラス。
    """

    def __init__(self, conn):
        """
        :param conn: Snowflakeコネクション（呼び出し元で生成）
        """
        self.conn = conn

    # ------------------------------------------------------------
    # 参照系（SELECT）
    # ------------------------------------------------------------
    def select(self, sql: str, params: tuple = None) -> list:
        """
        SELECTを実行し、結果をリスト（辞書形式）で返す。
        :param sql: 実行するSQL
        :param params: バインド変数（オプション）
        :return: 行のリスト（各行はカラム名→値のdict）
        """
        with self.conn.cursor(snowflake.connector.DictCursor) as cur:
            if params:
                cur.execute(sql, params)
            else:
                cur.execute(sql)
            return cur.fetchall()

    # ------------------------------------------------------------
    # 更新系（INSERT/UPDATE/DELETE）
    # ------------------------------------------------------------
    def execute(self, sql: str, params: tuple = None) -> int:
        """
        INSERT/UPDATE/DELETEを実行し、影響行数を返す。
        :param sql: 実行するSQL
        :param params: バインド変数（オプション）
        :return: 更新行数
        """
        with self.conn.cursor() as cur:
            if params:
                cur.execute(sql, params)
            else:
                cur.execute(sql)
            return cur.rowcount

    # ------------------------------------------------------------
    # バッチ挿入（効率的な大量INSERT）
    # ------------------------------------------------------------
    def bulk_insert(self, table_name: str, columns: list, rows: list) -> int:
        """
        指定テーブルにバルクINSERTを実行する。
        :param table_name: 'データベース.スキーマ.テーブル名'
        :param columns: カラム名リスト
        :param rows: 値のリスト（タプルのリスト）
        :return: 挿入行数
        """
        placeholders = ','.join(['%s'] * len(columns))
        col_names = ','.join(columns)
        sql = f"INSERT INTO {table_name} ({col_names}) VALUES ({placeholders})"
        with self.conn.cursor() as cur:
            cur.executemany(sql, rows)
            return cur.rowcount

    # ------------------------------------------------------------
    # トランザクション制御
    # ------------------------------------------------------------
    def begin(self):
        """トランザクション開始（自動コミットOFF）"""
        self.conn.autocommit(False)

    def commit(self):
        """コミット"""
        self.conn.commit()

    def rollback(self):
        """ロールバック"""
        self.conn.rollback()

    # ------------------------------------------------------------
    # 簡易CRUDラッパー（パラメータ考慮版）
    # ------------------------------------------------------------
    def insert_one(self, table: str, data: dict) -> int:
        """
        1行INSERT（辞書形式）
        :param table: 'DB.SCHEMA.TABLE'
        :param data: {カラム名: 値, ...}
        :return: 影響行数
        """
        columns = list(data.keys())
        values = tuple(data.values())
        placeholders = ','.join(['%s'] * len(columns))
        sql = f"INSERT INTO {table} ({','.join(columns)}) VALUES ({placeholders})"
        return self.execute(sql, values)

    def update(self, table: str, data: dict, where: str, where_params: tuple) -> int:
        """
        更新処理
        :param table: 'DB.SCHEMA.TABLE'
        :param data: {カラム名: 値, ...}
        :param where: WHERE句（プレースホルダ含む）
        :param where_params: WHERE用パラメータ
        :return: 影響行数
        """
        set_clause = ','.join([f"{k}=%s" for k in data.keys()])
        values = tuple(data.values()) + where_params
        sql = f"UPDATE {table} SET {set_clause} WHERE {where}"
        return self.execute(sql, values)

    def delete(self, table: str, where: str, where_params: tuple) -> int:
        """
        削除処理
        :param table: 'DB.SCHEMA.TABLE'
        :param where: WHERE句
        :param where_params: パラメータ
        :return: 影響行数
        """
        sql = f"DELETE FROM {table} WHERE {where}"
        return self.execute(sql, where_params)
```

3. tests/test_crud.py（テスト実行用スクリプト）

```python
# -*- coding: utf-8 -*-
"""
SqlRunnerとSqlReaderの動作確認を行うテストスクリプト。
実際にSnowflakeに接続して実行する。
"""
import snowflake.connector
from common.sql_reader import SqlReader
from common.sql_runner import SqlRunner

# ▼ 接続情報は環境変数などに外出し推奨
conn = snowflake.connector.connect(
    user='YOUR_USER',
    password='YOUR_PASSWORD',
    account='YOUR_ACCOUNT',
    warehouse='YOUR_WAREHOUSE',
    database='TEST_DB',
    schema='TEST_SCHEMA'
)

runner = SqlRunner(conn)

# 1. ステージからCREATE TABLE文を読んで実行
create_sql = SqlReader.read_sql_from_stage(conn, '@MY_STAGE/sql/create_test_table.sql')
print("【CREATE】")
runner.execute(create_sql)

# 2. ステージからINSERT文を読んで実行（データ投入）
insert_sql = SqlReader.read_sql_from_stage(conn, '@MY_STAGE/sql/insert_test_data.sql')
print("【INSERT】")
runner.execute(insert_sql)

# 3. SELECT確認
print("【SELECT】")
rows = runner.select("SELECT * FROM TEST_DB.TEST_SCHEMA.EMPLOYEE")
for r in rows:
    print(r)

# 4. UPDATE
print("【UPDATE】")
runner.update(
    table='TEST_DB.TEST_SCHEMA.EMPLOYEE',
    data={'name': '田中更新'},
    where='id = %s',
    where_params=(1,)
)

# 5. トランザクション & ロールバック確認
print("【トランザクション: 開始→INSERT→ロールバック】")
runner.begin()
runner.insert_one('TEST_DB.TEST_SCHEMA.EMPLOYEE', {'id': 99, 'name': 'ロールバック用'})
runner.rollback()
print("ロールバック後（id=99は存在しないはず）:", runner.select("SELECT * FROM EMPLOYEE WHERE id=99"))

# 6. バルクインサート
print("【バルクインサート】")
bulk_data = [(10, '一括1'), (11, '一括2')]
runner.bulk_insert(
    table_name='TEST_DB.TEST_SCHEMA.EMPLOYEE',
    columns=['id', 'name'],
    rows=bulk_data
)

# 7. 削除
print("【DELETE】")
runner.delete('TEST_DB.TEST_SCHEMA.EMPLOYEE', 'id = %s', (10,))

conn.close()
```
