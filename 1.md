	•	Stage SQL 文件
	•	测试表 SQL
	•	测试数据
	•	Streamlit 画面测试
	•	完整文件结构
	•	每个模块的日文说明

全部代码都能在 Snowflake 内部 Python Runtime 运行（直接使用 session）。

五、SqlReader 共通类

文件：

common/sql_reader.py

代码：

import hashlib
import re
from typing import Dict


class SqlReader:
    """
    SQLファイルをStageから読み込み、
    キャッシュ管理・MD5チェック・HotReloadを行う共通クラス
    """

    def __init__(self, session, stage_name: str):
        """
        初期化

        Parameters
        ----------
        session : Snowpark Session
        stage_name : str
            例 @MAIN_H.GOLD.SQL_STAGE
        """
        self.session = session
        self.stage_name = stage_name

        # SQLキャッシュ
        self.sql_cache: Dict[str, Dict] = {}

    def _calc_md5(self, content: str) -> str:
        """MD5計算"""
        return hashlib.md5(content.encode("utf-8")).hexdigest()

    def _read_stage_file(self, filename: str) -> str:
        """
        Stage から直接ファイルを読む
        """

        sql = f"""
        SELECT $1
        FROM @{self.stage_name}/{filename}
        """

        df = self.session.sql(sql).collect()

        content = "\n".join([row[0] for row in df])

        return content

    def _parse_sql(self, content: str) -> Dict[str, str]:
        """
        SQLファイルを解析

        フォーマット
        -- name: query_name
        SQL文
        """

        sql_dict = {}

        pattern = r"-- name:\s*(\w+)([\s\S]*?)(?=-- name:|\Z)"

        matches = re.findall(pattern, content)

        for name, sql in matches:
            sql_dict[name.strip()] = sql.strip()

        return sql_dict

    def get_sql(self, filename: str, sql_name: str) -> str:
        """
        SQL取得

        HotReload + MD5チェック付き
        """

        content = self._read_stage_file(filename)

        md5 = self._calc_md5(content)

        cache = self.sql_cache.get(filename)

        # MD5違えば再読み込み
        if cache is None or cache["md5"] != md5:

            parsed = self._parse_sql(content)

            self.sql_cache[filename] = {
                "md5": md5,
                "sql_map": parsed
            }

        return self.sql_cache[filename]["sql_map"][sql_name]

六、SqlRunner 共通类

文件：

common/sql_runner.py

代码：

from typing import Dict, List


class SqlRunner:
    """
    SQL実行共通クラス

    CRUD + トランザクション管理
    """

    def __init__(self, session, reader):
        """
        Parameters
        ----------
        session : Snowpark Session
        reader : SqlReader
        """

        self.session = session
        self.reader = reader

    def select(self, filename: str, sql_name: str, params: Dict = None):
        """
        SELECT 実行
        """

        sql = self.reader.get_sql(filename, sql_name)

        df = self.session.sql(sql, params)

        return df.collect()

    def execute(self, filename: str, sql_name: str, params: Dict = None):
        """
        INSERT / UPDATE / DELETE

        影響件数を返す
        """

        sql = self.reader.get_sql(filename, sql_name)

        result = self.session.sql(sql, params).collect()

        if result:
            return result[0][0]

        return 0

    def execute_many(self, filename: str, sql_name: str, param_list: List[Dict]):
        """
        バッチ処理
        """

        total = 0

        for p in param_list:
            total += self.execute(filename, sql_name, p)

        return total

    def transaction(self, func):
        """
        トランザクション実行
        """

        try:

            self.session.sql("BEGIN").collect()

            func(self)

            self.session.sql("COMMIT").collect()

        except Exception as e:

            self.session.sql("ROLLBACK").collect()

            raise e

七、Streamlit 测试画面

文件

app/streamlit_app.py

代码

import streamlit as st
from snowflake.snowpark.context import get_active_session

from common.sql_reader import SqlReader
from common.sql_runner import SqlRunner


session = get_active_session()

reader = SqlReader(session, "@MAIN_H.GOLD.SQL_STAGE")
runner = SqlRunner(session, reader)

st.title("Snowflake SQL Framework Test")

menu = st.sidebar.selectbox(
    "Menu",
    ["SELECT", "INSERT", "UPDATE", "DELETE", "TRANSACTION"]
)

if menu == "SELECT":

    if st.button("select all"):

        rows = runner.select("user_sql.sql", "select_all")

        st.write(rows)

if menu == "INSERT":

    user_id = st.number_input("USER_ID")
    name = st.text_input("NAME")
    email = st.text_input("EMAIL")

    if st.button("insert"):

        count = runner.execute(
            "user_sql.sql",
            "insert_user",
            {
                "user_id": user_id,
                "user_name": name,
                "email": email
            }
        )

        st.success(f"insert count={count}")

if menu == "UPDATE":

    user_id = st.number_input("USER_ID")
    name = st.text_input("NAME")
    email = st.text_input("EMAIL")

    if st.button("update"):

        count = runner.execute(
            "user_sql.sql",
            "update_user",
            {
                "user_id": user_id,
                "user_name": name,
                "email": email
            }
        )

        st.success(f"update count={count}")

if menu == "DELETE":

    user_id = st.number_input("USER_ID")

    if st.button("delete"):

        count = runner.execute(
            "user_sql.sql",
            "delete_user",
            {
                "user_id": user_id
            }
        )

        st.success(f"delete count={count}")

if menu == "TRANSACTION":

    if st.button("transaction test"):

        def tx(r):

            r.execute("user_sql.sql", "insert_user",
                      {"user_id": 100,
                       "user_name": "TX_TEST",
                       "email": "tx@test.com"})

            # 故意エラー
            raise Exception("force rollback")

        try:

            runner.transaction(tx)

        except:

            st.error("rollback executed")
