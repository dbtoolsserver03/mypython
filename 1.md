五、SqlReader（生产级）

common/sql_reader.py

"""
SqlReader 共通クラス

【目的】
Stageに保存されているSQLファイルを読み込み、
SQLをキャッシュ管理しながら提供する。

【機能】
1 SQLファイルの読み込み
2 SQLのnameタグ解析
3 MD5による変更検知
4 キャッシュ管理
5 Hot Reload対応

SQL形式例

-- name: select_user
SELECT * FROM TABLE WHERE ID=:id

-- name: insert_user
INSERT INTO ...
"""

import hashlib
import re
from typing import Dict
from snowflake.snowpark.context import get_active_session


class SqlReader:
    """
    SQL読み込み共通クラス
    """

    def __init__(self, stage_name: str):
        """
        コンストラクタ

        Parameters
        ----------
        stage_name : str
            SQLファイルを格納しているStage名
        """

        self.session = get_active_session()
        self.stage = stage_name

        # SQLキャッシュ
        self.sql_cache: Dict[str, Dict] = {}

    def _calc_md5(self, text: str) -> str:
        """
        SQL文字列からMD5を生成する
        """

        return hashlib.md5(text.encode("utf-8")).hexdigest()

    def _read_stage_file(self, filename: str) -> str:
        """
        StageからSQLファイルを直接読み込む

        Snowflakeでは GET を使わず
        SELECT で Stage ファイルを読む
        """

        query = f"""
        SELECT $1
        FROM @{self.stage}/{filename}
        """

        df = self.session.sql(query).collect()

        content = "\n".join([row[0] for row in df])

        return content

    def _parse_sql(self, content: str) -> Dict[str, str]:
        """
        SQLファイル解析

        -- name: xxx
        をキーにSQLを分割する
        """

        pattern = r"-- name:\s*(\w+)"
        parts = re.split(pattern, content)

        sql_map = {}

        for i in range(1, len(parts), 2):
            name = parts[i]
            sql = parts[i + 1].strip()

            sql_map[name] = sql

        return sql_map

    def load_sql_file(self, filename: str):
        """
        SQLファイルをロードする
        """

        content = self._read_stage_file(filename)

        md5 = self._calc_md5(content)

        # 変更なしならスキップ
        if filename in self.sql_cache:
            if self.sql_cache[filename]["md5"] == md5:
                return

        sql_map = self._parse_sql(content)

        self.sql_cache[filename] = {
            "md5": md5,
            "sql": sql_map
        }

    def get_sql(self, filename: str, sql_name: str) -> str:
        """
        SQL取得
        """

        self.load_sql_file(filename)

        return self.sql_cache[filename]["sql"][sql_name]

六、SqlRunner（执行共通）

common/sql_runner.py

"""
SqlRunner 共通クラス

【役割】

SQLの実行を統一管理する

機能
1 SELECT
2 INSERT
3 UPDATE
4 DELETE
5 BATCH処理
6 トランザクション管理
"""

from typing import Dict, List
from snowflake.snowpark.context import get_active_session


class SqlRunner:

    def __init__(self, sql_reader):

        self.session = get_active_session()
        self.reader = sql_reader

    def query(self, file, name, params: Dict = None):
        """
        SELECT専用

        Returns
        -------
        list
        """

        sql = self.reader.get_sql(file, name)

        df = self.session.sql(sql, params)

        return df.collect()

    def execute(self, file, name, params: Dict = None):
        """
        INSERT UPDATE DELETE

        Returns
        -------
        影響行数
        """

        sql = self.reader.get_sql(file, name)

        result = self.session.sql(sql, params).collect()

        return result[0][0] if result else 0

    def batch_execute(self, file, name, param_list: List[Dict]):
        """
        バッチ実行
        """

        sql = self.reader.get_sql(file, name)

        count = 0

        for p in param_list:
            self.session.sql(sql, p).collect()
            count += 1

        return count

    def transaction(self, func):
        """
        トランザクション管理
        """

        try:

            self.session.sql("BEGIN").collect()

            func()

            self.session.sql("COMMIT").collect()

        except Exception as e:

            self.session.sql("ROLLBACK").collect()

            raise e

七、Streamlit测试画面

app/streamlit_app.py

import streamlit as st
from common.sql_reader import SqlReader
from common.sql_runner import SqlRunner

reader = SqlReader("SQL_STAGE")
runner = SqlRunner(reader)

st.title("Snowpark SQL 共通テスト")

menu = st.sidebar.selectbox(
    "機能",
    ["SELECT", "INSERT", "UPDATE", "DELETE", "TRANSACTION"]
)

if menu == "SELECT":

    data = runner.query("user_sql.sql", "select_all")

    st.write(data)

if menu == "INSERT":

    name = st.text_input("名前")
    email = st.text_input("メール")

    if st.button("追加"):

        runner.execute(
            "user_sql.sql",
            "insert_user",
            {"user_name": name, "email": email}
        )

        st.success("INSERT成功")

if menu == "UPDATE":

    uid = st.number_input("ID")

    name = st.text_input("名前")

    email = st.text_input("メール")

    if st.button("更新"):

        runner.execute(
            "user_sql.sql",
            "update_user",
            {
                "user_id": uid,
                "user_name": name,
                "email": email
            }
        )

        st.success("UPDATE成功")

if menu == "DELETE":

    uid = st.number_input("削除ID")

    if st.button("削除"):

        runner.execute(
            "user_sql.sql",
            "delete_user",
            {"user_id": uid}
        )

        st.success("DELETE成功")

if menu == "TRANSACTION":

    def tx():

        runner.execute(
            "user_sql.sql",
            "insert_user",
            {"user_name": "TX", "email": "tx@test.com"}
        )

        raise Exception("テストロールバック")

    if st.button("TXテスト"):

        try:
            runner.transaction(tx)
        except:
            st.error("ROLLBACK確認")
