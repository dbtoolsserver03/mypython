この設計では、接続の管理、SQLの外部管理（Stage）、トランザクション制御、エラーハンドリングに重点を置いています。

3. 共通ユーティリティ (snowpark_utils.py)
ここがコアとなる実装です。SqlReader と SqlRunner を実装しています。
import logging
from snowflake.snowpark import Session, DataFrame
from snowflake.snowpark.exceptions import SnowparkClientException
from snowflake.snowpark.functions import col

# ログ設定
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class SqlReader:
    """
    Snowflake Stage上のSQLファイルを読み込むクラス
    """
    @staticmethod
    def read_from_stage(session: Session, stage_path: str) -> str:
        """
        Stage上のファイルをストリームとして読み込み、文字列として返す
        例: @my_stage/queries/select_data.sql
        """
        try:
            logger.info(f"Reading SQL from stage: {stage_path}")
            # get_streamでファイルを読み込む
            stream = session.file.get_stream(stage_path)
            sql_content = stream.read().decode('utf-8')
            return sql_content
        except Exception as e:
            logger.error(f"Failed to read SQL from stage: {e}")
            raise

class SqlRunner:
    """
    SQLの実行、CRUD、トランザクション、一括処理を管理するクラス
    """
    def __init__(self, session: Session):
        self.session = session

    def execute_query(self, sql_text: str) -> DataFrame:
        """SQLを直接実行し、DataFrameを返す"""
        try:
            return self.session.sql(sql_text)
        except Exception as e:
            logger.error(f"SQL Execution Error: {e}")
            raise

    def fetch_as_list(self, sql_text: str):
        """SQLを実行し、結果を辞書形式のリストで返す（キャッシュ的な利用）"""
        return self.execute_query(sql_text).collect()

    def batch_insert(self, df: DataFrame, table_name: str, mode: str = "append"):
        """大量データを一括挿入する"""
        try:
            df.write.mode(mode).save_as_table(table_name)
            logger.info(f"Successfully inserted data into {table_name}")
        except Exception as e:
            logger.error(f"Batch Insert Error: {e}")
            raise

    def transaction_block(self, operations_func):
        """
        トランザクション管理を行う高階関数
        operations_func: Sessionを受け取り、処理を行う関数
        """
        try:
            self.session.sql("BEGIN").collect()
            logger.info("Transaction Started")
            
            # 外部から渡された一連の処理を実行
            result = operations_func(self.session)
            
            self.session.sql("COMMIT").collect()
            logger.info("Transaction Committed")
            return result
        except Exception as e:
            self.session.sql("ROLLBACK").collect()
            logger.error(f"Transaction Rolled back due to: {e}")
            raise

    def update_record(self, table_name: str, set_values: dict, condition: str):
        """簡易的なUpdate（Snowpark DataFrame APIを使用）"""
        target_table = self.session.table(table_name)
        target_table.update(set_values, condition)
        logger.info(f"Updated {table_name} where {condition}")

    def delete_record(self, table_name: str, condition: str):
        """簡易的なDelete"""
        target_table = self.session.table(table_name)
        target_table.delete(condition)
        logger.info(f"Deleted from {table_name} where {condition}")

4. 信頼性テスト用スクリプト (main_test.py)
共通部品が正しく動作するか、CRUD、トランザクション、ロールバックのシナリオでテストします。
import toml
from snowflake.snowpark import Session
from utils.snowpark_utils import SqlReader, SqlRunner
import pandas as pd

def main():
    # 1. 接続の確立
    config = toml.load(".streamlit/secrets.toml")["snowflake"]
    session = Session.builder.configs(config).create()
    runner = SqlRunner(session)
    
    table_name = "TEST_SALES"

    try:
        print("--- [1. Setup: テストデータの準備] ---")
        data = pd.DataFrame([
            {"ID": 1, "PRODUCT": "Laptop", "PRICE": 1200},
            {"ID": 2, "PRODUCT": "Mouse", "PRICE": 25}
        ])
        snow_df = session.create_dataframe(data)
        runner.batch_insert(snow_df, table_name, mode="overwrite")

        print("--- [2. Transaction & Rollback Test] ---")
        def failed_transaction(sess):
            # 正常な更新
            sess.sql(f"UPDATE {table_name} SET PRICE = 999 WHERE ID = 1").collect()
            print("Update 1 applied...")
            # 意図的なエラー発生（存在しないカラムへのアクセス）
            sess.sql(f"UPDATE {table_name} SET INVALID_COL = 0").collect()

        try:
            runner.transaction_block(failed_transaction)
        except:
            print("Expected Error caught. Checking if rollback worked...")
            # ロールバック確認: ID=1の価格が1200のままであること
            res = session.table(table_name).filter(session.table(table_name).ID == 1).collect()
            print(f"Current Price for ID 1 (should be 1200): {res[0]['PRICE']}")

        print("--- [3. SqlReader Test (Stage)] ---")
        # 事前に SQLファイルをStageに置いておく必要があります
        # session.sql("CREATE OR REPLACE STAGE MY_STAGE").collect()
        # session.file.put("sql_files/test_query.sql", "@MY_STAGE", auto_compress=False)
        
        # sql_content = SqlReader.read_from_stage(session, "@MY_STAGE/test_query.sql")
        # print(f"SQL from Stage: {sql_content}")

        print("--- [4. CRUD: Update & Delete] ---")
        runner.update_record(table_name, {"PRICE": 1500}, "ID = 1")
        runner.delete_record(table_name, "ID = 2")
        
        session.table(table_name).show()

    finally:
        session.close()

if __name__ == "__main__":
    main()

5. 日本語による解説
① SqlReader クラス
 * 目的: SQL文をソースコード内にハードコーディングせず、Snowflakeの**Stage（内部/外部）**から管理するために使用します。
 * 利点: SQLを変更する際、Pythonコードをデプロイし直す必要がなく、Stage上のファイルを差し替えるだけで済みます。
② SqlRunner クラス
 * execute_query: 基本的なSQL実行です。
 * batch_insert: df.write を使用し、大量データを高速にテーブルへ同期します（本番環境でのデータロードに必須）。
 * transaction_block: 重要機能です。このブロック内で行われる複数の処理のうち、一つでも失敗すれば自動的に ROLLBACK し、データの整合性を守ります。
 * update/delete: Snowparkの型安全なDataFrame APIを利用して、安全にレコード操作を行います。
③ 信頼性テストのポイント
 * テストコードでは「わざとエラーを起こすトランザクション」を実行しています。
 * エラー発生時に ROLLBACK が走り、最初の更新（価格変更）が無効化されることを証明することで、本番環境での安全性を確認しています。
