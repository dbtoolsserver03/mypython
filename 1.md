5. Logger（企業では必須）

common/logger.py

import logging

def get_logger(name:str):

    """
    共通ログ生成
    """

    logger = logging.getLogger(name)

    if not logger.handlers:

        handler = logging.StreamHandler()

        formatter = logging.Formatter(
            "%(asctime)s | %(levelname)s | %(message)s"
        )

        handler.setFormatter(formatter)

        logger.addHandler(handler)

        logger.setLevel(logging.INFO)

    return logger

6. SqlReader（Stage から SQL 取得）

common/sql_reader.py

from snowflake.snowpark import Session

class SqlReader:

    """
    Stage に保存されている SQL ファイルを読み込む共通クラス

    特徴
    ・Snowflake 内部実行
    ・ローカル保存不要
    ・SQL 名指定取得
    """

    def __init__(self,session:Session,stage_file:str):

        """
        Parameters
        ----------
        session
            Snowpark セッション

        stage_file
            Stage 上の SQL ファイル

            例
            @COMMON_STAGE/sql/product_sql.sql
        """

        self.session=session
        self.stage_file=stage_file
        self.sql_map={}

        self._load_sql()

    def _load_sql(self):

        """
        SQL ファイル読み込み
        """

        df=self.session.sql(f"""
        SELECT $1
        FROM {self.stage_file}
        """)

        lines=[row[0] for row in df.collect()]

        current=None
        buffer=[]

        for line in lines:

            if line.strip().startswith("-- name:"):

                if current:

                    self.sql_map[current]="\n".join(buffer).strip()
                    buffer=[]

                current=line.split(":")[1].strip()

            else:

                buffer.append(line)

        if current:

            self.sql_map[current]="\n".join(buffer).strip()

    def get_sql(self,name:str):

        """
        SQL 名から SQL 文を取得
        """

        if name not in self.sql_map:

            raise Exception(f"SQL not found: {name}")

        return self.sql_map[name]

7. SqlRunner（SQL 実行共通）

common/sql_runner.py

from snowflake.snowpark import Session
from common.logger import get_logger

class SqlRunner:

    """
    SQL 実行共通クラス

    機能
    ・CRUD
    ・Batch Insert
    ・Transaction
    ・Rollback
    ・Query
    """

    def __init__(self,session:Session):

        self.session=session
        self.logger=get_logger("SqlRunner")

    def query(self,sql:str,params:dict=None):

        """
        SELECT 実行
        """

        self.logger.info("QUERY EXECUTE")

        df=self.session.sql(sql,params)

        return df.collect()

    def execute(self,sql:str,params:dict=None):

        """
        INSERT UPDATE DELETE
        """

        self.logger.info("EXECUTE SQL")

        self.session.sql(sql,params).collect()

    def batch_insert(self,sql:str,data:list):

        """
        バッチINSERT
        """

        self.logger.info("BATCH INSERT START")

        for row in data:

            self.session.sql(sql,row).collect()

        self.logger.info("BATCH INSERT END")

    def begin(self):

        self.logger.info("BEGIN TRANSACTION")

        self.session.sql("BEGIN").collect()

    def commit(self):

        self.logger.info("COMMIT")

        self.session.sql("COMMIT").collect()

    def rollback(self):

        self.logger.info("ROLLBACK")

        self.session.sql("ROLLBACK").collect()

from common.sql_reader import SqlReader
from common.sql_runner import SqlRunner


def main(session):

    print("===== TEST START =====")

    reader=SqlReader(
        session,
        "@COMMON_STAGE/sql/product_sql.sql"
    )

    runner=SqlRunner(session)

    """
    1 SELECT テスト
    """

    print("SELECT TEST")

    rows=runner.query(reader.get_sql("select_all"))

    for r in rows:
        print(r)


    """
    2 INSERT テスト
    """

    print("INSERT TEST")

    runner.execute(
        reader.get_sql("insert_product"),
        {"name":"Coffee","price":300,"category":"Drink"}
    )


    """
    3 UPDATE テスト
    """

    print("UPDATE TEST")

    runner.execute(
        reader.get_sql("update_product"),
        {
            "product_id":1,
            "name":"Apple Premium",
            "price":180,
            "category":"Fruit"
        }
    )


    """
    4 DELETE テスト
    """

    print("DELETE TEST")

    runner.execute(
        reader.get_sql("delete_product"),
        {"product_id":2}
    )


    """
    5 BATCH INSERT
    """

    print("BATCH INSERT TEST")

    runner.batch_insert(
        reader.get_sql("insert_product"),
        [
            {"name":"Tea","price":220,"category":"Drink"},
            {"name":"Cola","price":210,"category":"Drink"},
            {"name":"Cake","price":400,"category":"Food"}
        ]
    )


    """
    6 TRANSACTION 成功テスト
    """

    print("TRANSACTION SUCCESS TEST")

    try:

        runner.begin()

        runner.execute(
            reader.get_sql("insert_product"),
            {"name":"Water","price":80,"category":"Drink"}
        )

        runner.execute(
            reader.get_sql("insert_product"),
            {"name":"Juice","price":250,"category":"Drink"}
        )

        runner.commit()

    except Exception as e:

        runner.rollback()
        print(e)


    """
    7 ROLLBACK テスト
    """

    print("ROLLBACK TEST")

    try:

        runner.begin()

        runner.execute(
            reader.get_sql("insert_product"),
            {"name":"ErrorTest","price":999,"category":"Test"}
        )

        # 強制エラー
        runner.execute(
            "INSERT INTO DEMO_DB.PUBLIC.NOT_EXIST VALUES(1)"
        )

        runner.commit()

    except Exception as e:

        print("ERROR 발생 → ROLLBACK")

        runner.rollback()


    print("===== TEST END =====")
