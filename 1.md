3. SqlReaderクラス (sql_reader.py)

```python
"""
SqlReaderクラス
SnowflakeステージからSQLファイルを読み込む共通処理
"""

import os
import logging
from pathlib import Path
from typing import Optional, Dict, Any
import snowflake.snowpark as snowpark
from datetime import datetime, timedelta

# ロギング設定
logger = logging.getLogger(__name__)

class SqlReader:
    """
    SQLファイル読込共通クラス
    ステージ上のSQLファイルを読み込み、キャッシュ機能を提供する
    """
    
    def __init__(self, session: snowpark.Session, cache_enabled: bool = True, cache_expiry: int = 3600):
        """
        コンストラクタ
        
        Args:
            session: Snowparkセッション
            cache_enabled: キャッシュ有効フラグ
            cache_expiry: キャッシュ有効期限（秒）
        """
        self.session = session
        self.cache_enabled = cache_enabled
        self.cache_expiry = cache_expiry
        self.sql_cache: Dict[str, Dict[str, Any]] = {}  # SQLキャッシュ {file_name: {content, timestamp}}
        logger.info("SqlReader initialized. cache_enabled: %s, cache_expiry: %d", cache_enabled, cache_expiry)
    
    def read_sql_from_stage(self, stage_path: str, file_name: str) -> str:
        """
        ステージからSQLファイルを読み込む
        
        Args:
            stage_path: ステージパス（例: @MY_STAGE/sql/）
            file_name: SQLファイル名（例: test.sql）
            
        Returns:
            str: SQL文の内容
            
        Raises:
            FileNotFoundError: ファイルが見つからない場合
            Exception: その他のエラー
        """
        cache_key = f"{stage_path}/{file_name}"
        
        # キャッシュチェック
        if self.cache_enabled and self._is_cache_valid(cache_key):
            logger.info("Cache hit for SQL file: %s", cache_key)
            return self.sql_cache[cache_key]["content"]
        
        try:
            # ステージからファイルを読み込み
            full_path = f"{stage_path}/{file_name}"
            logger.info("Reading SQL file from stage: %s", full_path)
            
            # Snowflakeファイル読み込み
            stage_file = self.session.file.get_stream(full_path)
            content = stage_file.read().decode('utf-8')
            stage_file.close()
            
            # キャッシュに保存
            if self.cache_enabled:
                self.sql_cache[cache_key] = {
                    "content": content,
                    "timestamp": datetime.now()
                }
                logger.debug("SQL file cached: %s", cache_key)
            
            logger.info("Successfully read SQL file: %s (%d characters)", file_name, len(content))
            return content
            
        except Exception as e:
            logger.error("Failed to read SQL file: %s - %s", file_name, str(e))
            raise
    
    def read_sql_from_local(self, file_path: str) -> str:
        """
        ローカルファイルからSQLを読み込む（テスト用）
        
        Args:
            file_path: ローカルファイルパス
            
        Returns:
            str: SQL文の内容
        """
        try:
            path = Path(file_path)
            if not path.exists():
                raise FileNotFoundError(f"SQL file not found: {file_path}")
            
            with open(path, 'r', encoding='utf-8') as f:
                content = f.read()
            
            logger.info("Successfully read local SQL file: %s", file_path)
            return content
            
        except Exception as e:
            logger.error("Failed to read local SQL file: %s - %s", file_path, str(e))
            raise
    
    def _is_cache_valid(self, cache_key: str) -> bool:
        """
        キャッシュの有効性をチェック
        
        Args:
            cache_key: キャッシュキー
            
        Returns:
            bool: キャッシュ有効ならTrue
        """
        if cache_key not in self.sql_cache:
            return False
        
        cache_entry = self.sql_cache[cache_key]
        age = datetime.now() - cache_entry["timestamp"]
        return age.total_seconds() < self.cache_expiry
    
    def clear_cache(self) -> None:
        """キャッシュをクリア"""
        self.sql_cache.clear()
        logger.info("SQL cache cleared")
    
    def remove_from_cache(self, file_name: str) -> bool:
        """
        特定のファイルをキャッシュから削除
        
        Args:
            file_name: 削除するファイル名
            
        Returns:
            bool: 削除成功ならTrue
        """
        if file_name in self.sql_cache:
            del self.sql_cache[file_name]
            logger.info("Removed from cache: %s", file_name)
            return True
        return False
```

4. SqlRunnerクラス (sql_runner.py)

```python
"""
SqlRunnerクラス
SQL実行共通処理（CRUD、トランザクション、バッチ処理）
"""

import logging
from typing import Optional, List, Dict, Any, Union
from datetime import datetime
import snowflake.snowpark as snowpark
from snowflake.snowpark import Session
from snowflake.snowpark.functions import col
import pandas as pd

# ロギング設定
logger = logging.getLogger(__name__)

class SqlRunner:
    """
    SQL実行共通クラス
    CRUD操作、トランザクション管理、バッチ処理を提供
    """
    
    def __init__(self, session: snowpark.Session):
        """
        コンストラクタ
        
        Args:
            session: Snowparkセッション
        """
        self.session = session
        self.transaction_active = False
        logger.info("SqlRunner initialized")
    
    # ==================== 基本実行メソッド ====================
    
    def execute_sql(self, sql: str, params: Optional[Dict[str, Any]] = None) -> Optional[snowpark.DataFrame]:
        """
        SQL文を実行する
        
        Args:
            sql: 実行するSQL文
            params: パラメータ（名前付きプレースホルダー用）
            
        Returns:
            Optional[snowpark.DataFrame]: SELECT文の場合は結果DataFrame、それ以外はNone
        """
        try:
            logger.debug("Executing SQL: %s", sql[:200] + "..." if len(sql) > 200 else sql)
            
            if params:
                # パラメータ置換
                for key, value in params.items():
                    placeholder = f":{key}"
                    if isinstance(value, str):
                        sql = sql.replace(placeholder, f"'{value}'")
                    else:
                        sql = sql.replace(placeholder, str(value))
            
            # SQL実行
            result = self.session.sql(sql).collect()
            
            # SELECT文の場合（結果がある場合）
            if result and len(result) > 0:
                df = self.session.sql(sql)
                logger.info("SQL executed successfully. Returned %d rows", df.count())
                return df
            else:
                logger.info("SQL executed successfully. No rows returned")
                return None
                
        except Exception as e:
            logger.error("SQL execution failed: %s", str(e))
            logger.error("Failed SQL: %s", sql)
            raise
    
    def execute_sql_file(self, sql_content: str) -> Optional[snowpark.DataFrame]:
        """
        SQLファイルの内容を実行する
        
        Args:
            sql_content: SQLファイルの内容
            
        Returns:
            Optional[snowpark.DataFrame]: 実行結果
        """
        # 複数のSQL文に分割（セミコロンで区切る）
        statements = [s.strip() for s in sql_content.split(';') if s.strip()]
        
        last_result = None
        for i, stmt in enumerate(statements):
            if stmt:
                logger.debug("Executing statement %d/%d", i+1, len(statements))
                last_result = self.execute_sql(stmt)
        
        return last_result
    
    # ==================== CRUD操作 ====================
    
    def select(self, table_name: str, columns: Optional[List[str]] = None, 
               where: Optional[str] = None, limit: Optional[int] = None) -> snowpark.DataFrame:
        """
        データ取得（Read）
        
        Args:
            table_name: テーブル名
            columns: 取得カラム（Noneの場合は全カラム）
            where: WHERE句
            limit: 取得件数制限
            
        Returns:
            snowpark.DataFrame: 取得結果
        """
        try:
            # テーブル参照
            df = self.session.table(table_name)
            
            # カラム選択
            if columns:
                df = df.select(*[col(c) for c in columns])
            
            # WHERE句
            if where:
                df = df.where(where)
            
            # LIMIT
            if limit:
                df = df.limit(limit)
            
            logger.info("SELECT from %s completed. Rows: %d", table_name, df.count())
            return df
            
        except Exception as e:
            logger.error("SELECT failed on %s: %s", table_name, str(e))
            raise
    
    def insert(self, table_name: str, data: Union[Dict[str, Any], List[Dict[str, Any]]]) -> int:
        """
        データ挿入（Create）
        
        Args:
            table_name: テーブル名
            data: 挿入データ（単一行のDictまたは複数行のList[Dict]）
            
        Returns:
            int: 挿入行数
        """
        try:
            # データをリストに変換
            if isinstance(data, dict):
                rows = [data]
            else:
                rows = data
            
            if not rows:
                logger.warning("No data to insert")
                return 0
            
            # DataFrame作成
            df = self.session.create_dataframe(rows)
            
            # テーブルに書き込み
            result = df.write.mode("append").save_as_table(table_name)
            
            row_count = len(rows)
            logger.info("INSERT into %s completed. Rows inserted: %d", table_name, row_count)
            return row_count
            
        except Exception as e:
            logger.error("INSERT failed on %s: %s", table_name, str(e))
            raise
    
    def update(self, table_name: str, set_values: Dict[str, Any], 
               where: Optional[str] = None) -> int:
        """
        データ更新（Update）
        
        Args:
            table_name: テーブル名
            set_values: 更新値の辞書 {column: value}
            where: WHERE句（Noneの場合は全件更新）
            
        Returns:
            int: 更新行数
        """
        try:
            # SET句の構築
            set_clause = ", ".join([f"{k} = ?" for k in set_values.keys()])
            
            # SQL文構築
            sql = f"UPDATE {table_name} SET {set_clause}"
            if where:
                sql += f" WHERE {where}"
            
            # パラメータ準備
            params = list(set_values.values())
            
            # 実行
            result = self.session.sql(sql, params).collect()
            
            # 更新行数取得
            rows_updated = result[0][0] if result else 0
            logger.info("UPDATE on %s completed. Rows updated: %d", table_name, rows_updated)
            return rows_updated
            
        except Exception as e:
            logger.error("UPDATE failed on %s: %s", table_name, str(e))
            raise
    
    def delete(self, table_name: str, where: Optional[str] = None) -> int:
        """
        データ削除（Delete）
        
        Args:
            table_name: テーブル名
            where: WHERE句（Noneの場合は全件削除）
            
        Returns:
            int: 削除行数
        """
        try:
            sql = f"DELETE FROM {table_name}"
            if where:
                sql += f" WHERE {where}"
            
            result = self.session.sql(sql).collect()
            
            rows_deleted = result[0][0] if result else 0
            logger.info("DELETE from %s completed. Rows deleted: %d", table_name, rows_deleted)
            return rows_deleted
            
        except Exception as e:
            logger.error("DELETE failed on %s: %s", table_name, str(e))
            raise
    
    # ==================== バッチ処理 ====================
    
    def bulk_insert(self, table_name: str, data: List[Dict[str, Any]], 
                    batch_size: int = 1000) -> int:
        """
        バッチ挿入（大量データ用）
        
        Args:
            table_name: テーブル名
            data: 挿入データリスト
            batch_size: 1回のバッチサイズ
            
        Returns:
            int: 総挿入行数
        """
        total_inserted = 0
        try:
            for i in range(0, len(data), batch_size):
                batch = data[i:i+batch_size]
                inserted = self.insert(table_name, batch)
                total_inserted += inserted
                logger.debug("Batch %d/%d inserted: %d rows", 
                           i//batch_size + 1, (len(data)-1)//batch_size + 1, inserted)
            
            logger.info("Bulk insert completed. Total rows: %d", total_inserted)
            return total_inserted
            
        except Exception as e:
            logger.error("Bulk insert failed at batch: %s", str(e))
            raise
    
    # ==================== トランザクション管理 ====================
    
    def begin_transaction(self) -> None:
        """トランザクション開始"""
        try:
            self.session.sql("BEGIN").collect()
            self.transaction_active = True
            logger.info("Transaction started")
        except Exception as e:
            logger.error("Failed to start transaction: %s", str(e))
            raise
    
    def commit(self) -> None:
        """コミット"""
        try:
            if self.transaction_active:
                self.session.sql("COMMIT").collect()
                self.transaction_active = False
                logger.info("Transaction committed")
        except Exception as e:
            logger.error("Failed to commit: %s", str(e))
            raise
    
    def rollback(self) -> None:
        """ロールバック"""
        try:
            if self.transaction_active:
                self.session.sql("ROLLBACK").collect()
                self.transaction_active = False
                logger.info("Transaction rolled back")
        except Exception as e:
            logger.error("Failed to rollback: %s", str(e))
            raise
    
    def execute_in_transaction(self, operations: List[Dict[str, Any]]) -> bool:
        """
        トランザクション内で複数操作を実行
        
        Args:
            operations: 操作リスト
                例: [
                    {"type": "insert", "table": "users", "data": {"name": "test"}},
                    {"type": "update", "table": "users", "set": {"status": "active"}, "where": "id=1"}
                ]
            
        Returns:
            bool: 成功ならTrue
        """
        try:
            self.begin_transaction()
            
            for op in operations:
                op_type = op.get("type")
                
                if op_type == "insert":
                    self.insert(op["table"], op["data"])
                elif op_type == "update":
                    self.update(op["table"], op["set"], op.get("where"))
                elif op_type == "delete":
                    self.delete(op["table"], op.get("where"))
                elif op_type == "sql":
                    self.execute_sql(op["sql"])
                else:
                    raise ValueError(f"Unknown operation type: {op_type}")
            
            self.commit()
            logger.info("Transaction completed successfully with %d operations", len(operations))
            return True
            
        except Exception as e:
            self.rollback()
            logger.error("Transaction failed, rolled back: %s", str(e))
            return False
    
    # ==================== 結果取得 ====================
    
    def to_pandas(self, df: snowpark.DataFrame) -> pd.DataFrame:
        """
        Snowpark DataFrameをPandas DataFrameに変換
        
        Args:
            df: Snowpark DataFrame
            
        Returns:
            pd.DataFrame: Pandas DataFrame
        """
        return df.to_pandas()
    
    def to_dict(self, df: snowpark.DataFrame) -> List[Dict[str, Any]]:
        """
        Snowpark DataFrameを辞書リストに変換
        
        Args:
            df: Snowpark DataFrame
            
        Returns:
            List[Dict[str, Any]]: 辞書リスト
        """
        return [row.as_dict() for row in df.collect()]
```

"""
ユーティリティ関数
"""

import logging
import toml
from pathlib import Path
from typing import Dict, Any
import snowflake.snowpark as snowpark
from snowflake.snowpark import Session

# ロギング設定
logging.basicConfig(
    level=logging.INFO,
    format='%(asasctime)s - %(name)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

def create_snowpark_session(secrets_path: str = ".secrets.toml") -> Session:
    """
    Snowparkセッションを作成
    
    Args:
        secrets_path: シークレットファイルパス
        
    Returns:
        Session: Snowparkセッション
    """
    try:
        # 設定読み込み
        with open(secrets_path, 'r') as f:
            secrets = toml.load(f)
        
        snowflake_config = secrets['snowflake']
        
        # セッション作成
        connection_params = {
            "account": snowflake_config["account"],
            "user": snowflake_config["user"],
            "password": snowflake_config["password"],
            "role": snowflake_config["role"],
            "warehouse": snowflake_config["warehouse"],
            "database": snowflake_config["database"],
            "schema": snowflake_config["schema"]
        }
        
        session = Session.builder.configs(connection_params).create()
        logger.info("Snowpark session created successfully")
        return session
        
    except Exception as e:
        logger.error("Failed to create Snowpark session: %s", str(e))
        raise

def close_session(session: Session) -> None:
    """
    セッションをクローズ
    
    Args:
        session: Snowparkセッション
    """
    if session:
        session.close()
        logger.info("Snowpark session closed")

10. 主な機能の説明（日本語）

SqlReaderクラス

· 目的: ステージ上のSQLファイルを読み込む共通処理
· 主な機能:
  · read_sql_from_stage(): ステージからSQLファイル読み込み
  · read_sql_from_local(): ローカルからSQLファイル読み込み（テスト用）
  · キャッシュ機能: 読み込んだSQLをメモリにキャッシュ
  · 有効期限管理: キャッシュの有効期限を設定可能

SqlRunnerクラス

· 目的: SQL実行の共通処理
· 主な機能:
  · execute_sql(): 任意のSQL実行
  · CRUD操作: select, insert, update, delete
  · バッチ処理: bulk_insert（大量データ挿入）
  · トランザクション管理: begin, commit, rollback
  · 複数操作のトランザクション実行: execute_in_transaction()

信頼性のポイント

1. エラーハンドリング: 各メソッドで例外を適切にキャッチ
2. ロギング: 詳細なログ出力で追跡可能
3. トランザクション: 一貫性を保証
4. キャッシュ: パフォーマンス最適化
5. バッチ処理: 大量データの効率的な処理

この共通処理を使用することで、Snowflakeでのデータ操作を安全かつ効率的に行うことができます。
