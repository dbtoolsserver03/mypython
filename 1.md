四、SqlReader（Stage读取 + MD5 + 缓存）

common/sql_reader.py

import hashlib
import re
from typing import Dict
from snowflake.snowpark import Session


class SqlReader:
    """
    SQLファイルをSnowflakeのStageから読み込む共通クラス

    機能
    -----------------------------
    1 StageからSQLファイルを直接読み込む
    2 SQLをname単位で解析
    3 MD5で変更検知
    4 メモリキャッシュ
    """

    def __init__(self, session: Session, stage_path: str):
        """
        Parameters
        ----------
        session : Snowpark Session
        stage_path : 例 '@DEMO_DB.PUBLIC.SQL_STAGE'
        """

        self.session = session
        self.stage_path = stage_path

        # SQLキャッシュ
        self.sql_cache: Dict[str, Dict[str, str]] = {}

        # MD5キャッシュ
        self.md5_cache: Dict[str, str] = {}

    def _read_stage_file(self, file_name: str) -> str:
        """
        StageからSQLファイルを読み込む
        """

        query = f"""
        SELECT $1
        FROM {self.stage_path}/{file_name}
        """

        rows = self.session.sql(query).collect()

        content = "\n".join([r[0] for r in rows])

        return content

    def _calc_md5(self, text: str) -> str:
        """
        SQL内容のMD5を計算
        """

        return hashlib.md5(text.encode("utf-8")).hexdigest()

    def _parse_sql(self, content: str) -> Dict[str, str]:
        """
        SQLファイルを解析

        形式

        -- name: select_user
        SELECT ...
        """

        sql_map = {}

        pattern = r"--\s*name:\s*(\w+)([\s\S]*?)(?=(--\s*name:|\Z))"

        matches = re.findall(pattern, content)

        for m in matches:
            name = m[0]
            sql = m[1].strip()

            sql_map[name] = sql

        return sql_map

    def load_sql(self, file_name: str):
        """
        SQLファイルロード
        MD5で変更検知
        """

        content = self._read_stage_file(file_name)

        md5 = self._calc_md5(content)

        if file_name in self.md5_cache:

            if self.md5_cache[file_name] == md5:
                return

        sql_map = self._parse_sql(content)

        self.sql_cache[file_name] = sql_map

        self.md5_cache[file_name] = md5

    def get_sql(self, file_name: str, sql_name: str) -> str:
        """
        SQL取得
        """

        if file_name not in self.sql_cache:
            self.load_sql(file_name)

        return self.sql_cache[file_name][sql_name]

五、SqlRunner（CRUD + 事务）

common/sql_runner.py

from typing import Dict, List
from snowflake.snowpark import Session


class SqlRunner:
    """
    SQL実行共通クラス

    機能
    -----------------------
    SELECT
    INSERT
    UPDATE
    DELETE
    batch insert
    transaction
    """

    def __init__(self, session: Session):

        self.session = session

    def query(self, sql: str, params: Dict = None):
        """
        SELECT実行
        """

        df = self.session.sql(sql, params)

        return df.collect()

    def execute(self, sql: str, params: Dict = None):
        """
        更新系SQL実行

        影響件数を返す
        """

        result = self.session.sql(sql, params).collect()

        if len(result) > 0:
            return result[0][0]

        return 0

    def batch_insert(self, sql: str, param_list: List[Dict]):
        """
        バッチINSERT
        """

        count = 0

        for p in param_list:
            count += self.execute(sql, p)

        return count

    def begin(self):

        self.session.sql("BEGIN").collect()

    def commit(self):

        self.session.sql("COMMIT").collect()

    def rollback(self):

        self.session.sql("ROLLBACK").collect()

    def transaction(self, func):

        try:

            self.begin()

            func()

            self.commit()

        except Exception as e:

            self.rollback()

            raise e

六、Streamlit 画面测试

app.py

import streamlit as st

from snowflake.snowpark.context import get_active_session

from common.sql_reader import SqlReader
from common.sql_runner import SqlRunner


session = get_active_session()

reader = SqlReader(session, "@DEMO_DB.PUBLIC.SQL_STAGE")

runner = SqlRunner(session)

st.title("Snowpark SQL 共通テスト")

menu = st.sidebar.selectbox(
    "メニュー",
    [
        "select_all",
        "insert",
        "update",
        "delete",
        "transaction_test"
    ]
)

if menu == "select_all":

    sql = reader.get_sql("user_sql.sql", "select_all")

    rows = runner.query(sql)

    st.write(rows)

if menu == "insert":

    name = st.text_input("name")
    email = st.text_input("email")

    if st.button("insert"):

        sql = reader.get_sql("user_sql.sql", "insert_user")

        count = runner.execute(
            sql,
            {
                "user_name": name,
                "email": email
            }
        )

        st.success(f"insert count={count}")

if menu == "update":

    id = st.number_input("id")

    name = st.text_input("name")

    email = st.text_input("email")

    if st.button("update"):

        sql = reader.get_sql("user_sql.sql", "update_user")

        count = runner.execute(
            sql,
            {
                "user_id": id,
                "user_name": name,
                "email": email
            }
        )

        st.success(f"update count={count}")

if menu == "delete":

    id = st.number_input("id")

    if st.button("delete"):

        sql = reader.get_sql("user_sql.sql", "delete_user")

        count = runner.execute(
            sql,
            {
                "user_id": id
            }
        )

        st.success(f"delete count={count}")

if menu == "transaction_test":

    def tx():

        sql = reader.get_sql("user_sql.sql", "insert_user")

        runner.execute(sql, {
            "user_name": "TX_TEST",
            "email": "tx@test.com"
        })

        raise Exception("force rollback")

    if st.button("run tx"):

        try:

            runner.transaction(tx)

        except:

            st.error("rollback success")
