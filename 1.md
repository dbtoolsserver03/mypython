	•	stage SQL 文件格式

	•	详细日文说明

所有代码均可 直接在 Snowflake 内部 Python 环境运行（使用 session，无需自己建立连接）。

5 SqlReader 共通类

common/sql_reader.py

import hashlib
import re
from snowflake.snowpark.files import SnowflakeFile


class SqlReader:
    """
    SQLファイルをSnowflake Stageから読み込む共通クラス

    機能
    ----------------------------------------
    ・StageからSQLファイルを直接読み込み
    ・SQL文を name で管理
    ・MD5で更新検知
    ・キャッシュ管理
    ・Hot Reload対応
    """

    def __init__(self, session, stage_path: str):
        """
        初期化

        Parameters
        ----------
        session : Snowpark Session
        stage_path : str
            例: @MAIN_H.GOLD.SQL_STAGE
        """
        self.session = session
        self.stage_path = stage_path

        self.cache = {}
        self.md5_cache = {}

    def _read_stage_file(self, filename: str) -> str:
        """
        StageからSQLファイルを読み込む
        """

        stage_file = f"{self.stage_path}/{filename}"

        with SnowflakeFile.open(stage_file) as f:
            df = f.readlines()

        content = "\n".join([str(row[0]) for row in df if row[0]])

        return content

    def _calc_md5(self, content: str):

        return hashlib.md5(content.encode("utf-8")).hexdigest()

    def _parse_sql(self, content: str):

        """
        SQLファイルを解析

        例
        -- name: select_all
        SELECT * FROM TABLE;

        戻り値
        {
            "select_all": "SELECT * ..."
        }
        """

        sql_map = {}

        pattern = r"--\s*name:\s*(\w+)\s*(.*?)((?=--\s*name:)|$)"

        matches = re.findall(pattern, content, re.S)

        for name, sql, _ in matches:

            sql_map[name.strip()] = sql.strip()

        return sql_map

    def load_sql_file(self, filename: str):
        """
        SQLファイル読み込み

        ・MD5チェック
        ・キャッシュ更新
        """

        content = self._read_stage_file(filename)

        md5 = self._calc_md5(content)

        if filename in self.md5_cache and self.md5_cache[filename] == md5:
            return

        sql_map = self._parse_sql(content)

        self.cache[filename] = sql_map
        self.md5_cache[filename] = md5

    def get_sql(self, filename: str, name: str):

        """
        SQL取得
        """

        if filename not in self.cache:
            self.load_sql_file(filename)

        return self.cache[filename][name]

6 SqlRunner 共通类

common/sql_runner.py

class SqlRunner:
    """
    SQL実行共通クラス

    機能
    ---------------------------
    ・CRUD
    ・トランザクション
    ・影響件数取得
    """

    def __init__(self, session):

        self.session = session

    def query(self, sql: str, params: dict = None):

        """
        SELECT用
        """

        df = self.session.sql(sql, params)

        return df.collect()

    def execute(self, sql: str, params: dict = None):

        """
        INSERT UPDATE DELETE

        戻り値
        影響件数
        """

        result = self.session.sql(sql, params).collect()

        if result:
            return result[0][0]

        return 0

    def begin(self):

        self.session.sql("BEGIN").collect()

    def commit(self):

        self.session.sql("COMMIT").collect()

    def rollback(self):

        self.session.sql("ROLLBACK").collect()

    def transaction(self, func):

        """
        トランザクション実行
        """

        try:

            self.begin()

            func()

            self.commit()

        except Exception as e:

            self.rollback()

            raise e

7 Streamlit 测试画面

app/streamlit_app.py

import streamlit as st
from snowflake.snowpark.context import get_active_session

from common.sql_reader import SqlReader
from common.sql_runner import SqlRunner


session = get_active_session()

reader = SqlReader(session, "@MAIN_H.GOLD.SQL_STAGE")
runner = SqlRunner(session)

SQL_FILE = "user_sql.sql"


st.title("Snowpark SQL Framework Test")

menu = st.sidebar.selectbox(
    "Menu",
    [
        "Select All",
        "Select By ID",
        "Insert",
        "Update",
        "Delete",
        "Transaction Test"
    ]
)

# -----------------------
# select all
# -----------------------

if menu == "Select All":

    sql = reader.get_sql(SQL_FILE, "select_all")

    rows = runner.query(sql)

    st.write(rows)


# -----------------------
# select by id
# -----------------------

if menu == "Select By ID":

    uid = st.number_input("User ID")

    sql = reader.get_sql(SQL_FILE, "select_by_id")

    rows = runner.query(sql, {"user_id": uid})

    st.write(rows)


# -----------------------
# insert
# -----------------------

if menu == "Insert":

    uid = st.number_input("User ID")
    name = st.text_input("Name")
    email = st.text_input("Email")

    if st.button("Insert"):

        sql = reader.get_sql(SQL_FILE, "insert_user")

        count = runner.execute(
            sql,
            {
                "user_id": uid,
                "user_name": name,
                "email": email
            }
        )

        st.success(f"Inserted: {count}")


# -----------------------
# update
# -----------------------

if menu == "Update":

    uid = st.number_input("User ID")
    name = st.text_input("Name")
    email = st.text_input("Email")

    if st.button("Update"):

        sql = reader.get_sql(SQL_FILE, "update_user")

        count = runner.execute(
            sql,
            {
                "user_id": uid,
                "user_name": name,
                "email": email
            }
        )

        st.success(f"Updated: {count}")


# -----------------------
# delete
# -----------------------

if menu == "Delete":

    uid = st.number_input("User ID")

    if st.button("Delete"):

        sql = reader.get_sql(SQL_FILE, "delete_user")

        count = runner.execute(sql, {"user_id": uid})

        st.success(f"Deleted: {count}")


# -----------------------
# transaction
# -----------------------

if menu == "Transaction Test":

    def tx():

        sql = reader.get_sql(SQL_FILE, "insert_user")

        runner.execute(
            sql,
            {
                "user_id": 999,
                "user_name": "TX_TEST",
                "email": "tx@test.com"
            }
        )

        raise Exception("simulate error")

    if st.button("Run Transaction"):

        try:

            runner.transaction(tx)

        except Exception:

            st.error("Transaction rollback success")

Stage SQL 読み込み
