
import re


class SqlRunner:
    """
    Snowflake(Snowpark) 用 SQL 実行共通クラス

    本クラスの役割
    ---------------------------------------
    1. SQLの実行（SELECT / INSERT / UPDATE / DELETE）
    2. 名前付きパラメータ (:param) を Snowpark 用に変換
    3. バッチ INSERT
    4. トランザクション管理
    5. CRUD 共通化

    Snowpark の session.sql() は
    「位置パラメータ(?) + list形式」のみサポートしている。

    しかし業務SQLでは通常

        :user_id
        :user_name

    のような「名前付きパラメータ」を使用するため、
    本クラスでは自動変換処理を行う。

    例

        SQL
        --------------------------------
        INSERT INTO USER_TEST
        VALUES(:user_id,:user_name)

        Python
        --------------------------------
        {
            "user_id":1,
            "user_name":"Taro"
        }

        ↓ 内部変換

        SQL
        --------------------------------
        INSERT INTO USER_TEST
        VALUES(?,?)

        PARAM
        --------------------------------
        [1,"Taro"]

    これによりアプリ側は
    名前付きパラメータ形式でSQLを記述できる。
    """

    def __init__(self, session):
        """
        コンストラクタ

        Parameters
        ----------
        session : Snowpark Session
            Snowflake の実行セッション
            Streamlit in Snowflake では通常

                from snowflake.snowpark.context import get_active_session

                session = get_active_session()

            で取得する
        """

        self.session = session

    def _convert_named_params(self, sql, params):
        """
        名前付きパラメータを Snowpark 用に変換する内部メソッド

        Snowpark の session.sql() は

            ? (プレースホルダ)

        のみ対応しているため、

            :user_id

        のような名前付きパラメータを

            ?

        に変換する。

        同時に dict 形式のパラメータを
        SQL 内の順序に従った list に変換する。

        Parameters
        ----------
        sql : str
            実行対象SQL

        params : dict
            名前付きパラメータ

            例
            {
                "user_id":1,
                "user_name":"Taro"
            }

        Returns
        -------
        tuple
            (変換後SQL, パラメータ配列)

        例

            SQL
            --------------------------------
            SELECT * FROM USER_TEST
            WHERE USER_ID = :user_id

            params
            --------------------------------
            {"user_id":1}

            return
            --------------------------------
            SELECT * FROM USER_TEST
            WHERE USER_ID = ?

            [1]
        """

        # パラメータが存在しない場合
        if not params:
            return sql, []

        # :param を検出する正規表現
        pattern = re.compile(r":(\w+)")

        # SQL内のパラメータ名を取得
        # 例
        # :user_id
        # :user_name
        keys = pattern.findall(sql)

        values = []

        # SQL内の順番に従って
        # dict から値を取り出す
        for k in keys:

            if k not in params:
                raise Exception(f"Missing param: {k}")

            values.append(params[k])

        # :param → ? に変換
        sql = pattern.sub("?", sql)

        return sql, values

    def query(self, sql, params=None):
        """
        SELECT 用メソッド

        SELECT 文を実行し結果を返す。

        Parameters
        ----------
        sql : str
            SQL文

        params : dict
            SQLパラメータ

        Returns
        -------
        list
            Snowpark Row の配列
        """

        # 名前付きパラメータ変換
        sql, values = self._convert_named_params(sql, params)

        # SQL実行
        df = self.session.sql(sql, values)

        # 結果取得
        return df.collect()

    def execute(self, sql, params=None):
        """
        INSERT / UPDATE / DELETE 用メソッド

        データ更新系SQLを実行する。

        Parameters
        ----------
        sql : str
            SQL文

        params : dict
            SQLパラメータ

        Returns
        -------
        int
            影響件数

        ※ Snowpark は影響件数を直接返さないため
        collect結果の件数を返す
        """

        sql, values = self._convert_named_params(sql, params)

        df = self.session.sql(sql, values)

        result = df.collect()

        return len(result)

    def batch_insert(self, sql, data_list):
        """
        バッチ INSERT

        複数データを一括登録する。

        Parameters
        ----------
        sql : str
            INSERT SQL

        data_list : list[dict]

            例

            [
                {"user_id":1,"user_name":"A"},
                {"user_id":2,"user_name":"B"}
            ]

        Returns
        -------
        int
            登録件数
        """

        count = 0

        for params in data_list:

            sql2, values = self._convert_named_params(sql, params)

            self.session.sql(sql2, values).collect()

            count += 1

        return count

    def transaction(self, func):
        """
        トランザクション処理

        BEGIN
        COMMIT
        ROLLBACK

        を自動制御する。

        使用例

            def tx(runner):

                runner.execute(sql1)
                runner.execute(sql2)

            runner.transaction(tx)

        処理フロー

        BEGIN
        ↓
        func実行
        ↓
        正常 → COMMIT
        例外 → ROLLBACK
        """

        try:

            # トランザクション開始
            self.session.sql("BEGIN").collect()

            result = func(self)

            # 正常終了
            self.session.sql("COMMIT").collect()

            return result

        except Exception as e:

            # エラー時ロールバック
            self.session.sql("ROLLBACK").collect()

            raise e

common/sql_reader.py

import hashlib
import time
from snowflake.snowpark.files import SnowflakeFile


class SqlReader:
    """
    SQLファイルをStageから読み込む共通クラス

    機能
    ----------------------------
    1 SQLファイル読み込み
    2 SQL解析（-- name: xxx）
    3 SQLキャッシュ
    4 MD5変更検知
    5 Hot Reload
    """

    _cache = {}

    def __init__(self, stage_path: str):
        """
        stage_path 例
        @MAIN_H.GOLD.SQL_STAGE/sql/user_sql.sql
        """
        self.stage_path = stage_path

    def _read_stage_file(self):

        """
        StageからSQLファイルを直接読み込む
        """

        with SnowflakeFile.open(self.stage_path, 'r') as f:
            content = f.read()

        return content

    def _calc_md5(self, text):

        """
        SQL内容のMD5を計算
        """

        return hashlib.md5(text.encode()).hexdigest()

    def _parse_sql(self, content):

        """
        SQLファイルを解析する

        -- name: xxx
        形式を辞書にする
        """

        sql_map = {}

        current_name = None
        sql_lines = []

        for line in content.splitlines():

            line = line.strip()

            if line.startswith("-- name:"):

                if current_name:
                    sql_map[current_name] = "\n".join(sql_lines)

                current_name = line.replace("-- name:", "").strip()

                sql_lines = []

            else:

                if line:
                    sql_lines.append(line)

        if current_name:
            sql_map[current_name] = "\n".join(sql_lines)

        return sql_map

    def load(self):

        """
        SQLをロード
        キャッシュ + MD5変更検知
        """

        content = self._read_stage_file()

        md5 = self._calc_md5(content)

        cache = SqlReader._cache.get(self.stage_path)

        if cache and cache["md5"] == md5:

            return cache["sql_map"]

        sql_map = self._parse_sql(content)

        SqlReader._cache[self.stage_path] = {
            "md5": md5,
            "sql_map": sql_map,
            "time": time.time()
        }

        return sql_map

    def get_sql(self, name):

        """
        指定SQLを取得
        """

        sql_map = self.load()

        if name not in sql_map:
            raise Exception(f"SQL not found: {name}")

        return sql_map[name]
