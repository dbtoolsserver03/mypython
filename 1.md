五、SqlReader 共通

src/common/sql_reader.py

from snowflake.snowpark import Session

class SqlReader:
    """
    SQLファイルをSnowflakeのStageから読み込む共通クラス
    """

    def __init__(self, session: Session, stage_path: str):
        """
        コンストラクタ

        Parameters
        ----------
        session : Snowpark Session
            Snowflakeのセッション

        stage_path : str
            SQLファイルが格納されているStage
            例: @DEMO_DB.PUBLIC.SQL_STAGE/user_sql.sql
        """
        self.session = session
        self.stage_path = stage_path
        self.sql_map = {}

        self._load_sql()

    def _load_sql(self):
        """
        StageからSQLファイルを読み込み
        -- name: xxx の形式でSQLを分割して辞書に格納
        """

        df = self.session.sql(f"SELECT $1 FROM {self.stage_path}")
        rows = df.collect()

        sql_text = "\n".join([r[0] for r in rows])

        current_key = None
        buffer = []

        for line in sql_text.split("\n"):

            if line.strip().startswith("-- name:"):
                if current_key:
                    self.sql_map[current_key] = "\n".join(buffer)

                current_key = line.split(":")[1].strip()
                buffer = []
            else:
                buffer.append(line)

        if current_key:
            self.sql_map[current_key] = "\n".join(buffer)

    def get_sql(self, name: str) -> str:
        """
        SQL取得

        Parameters
        ----------
        name : str
            SQL名

        Returns
        -------
        str
            SQL文
        """
        return self.sql_map.get(name)

六、SqlRunner 共通

src/common/sql_runner.py

from snowflake.snowpark import Session

class SqlRunner:
    """
    SQL実行共通クラス
    CRUD操作、トランザクション制御を提供
    """

    def __init__(self, session: Session):
        """
        コンストラクタ

        Parameters
        ----------
        session : Snowpark Session
        """
        self.session = session

    def select(self, sql: str, params: dict = None):
        """
        SELECT実行

        Returns
        -------
        list
        """

        df = self.session.sql(sql, params)
        return df.collect()

    def insert(self, sql: str, params: dict):
        """
        INSERT実行
        """
        self.session.sql(sql, params).collect()

    def update(self, sql: str, params: dict):
        """
        UPDATE実行
        """
        self.session.sql(sql, params).collect()

    def delete(self, sql: str, params: dict):
        """
        DELETE実行
        """
        self.session.sql(sql, params).collect()

    def batch_insert(self, sql: str, param_list: list):
        """
        バッチINSERT
        """

        for params in param_list:
            self.session.sql(sql, params).collect()

    def begin(self):
        """
        トランザクション開始
        """
        self.session.sql("BEGIN").collect()

    def commit(self):
        """
        コミット
        """
        self.session.sql("COMMIT").collect()

    def rollback(self):
        """
        ロールバック
        """
        self.session.sql("ROLLBACK").collect()

七、测试程序

src/app/test_app.py

from snowflake.snowpark import Session
from common.sql_reader import SqlReader
from common.sql_runner import SqlRunner

def main(session: Session):

    reader = SqlReader(
        session,
        "@DEMO_DB.PUBLIC.SQL_STAGE/user_sql.sql"
    )

    runner = SqlRunner(session)

    print("===== SELECT ALL =====")

    rows = runner.select(reader.get_sql("select_all"))

    for r in rows:
        print(r)

    print("===== INSERT =====")

    runner.insert(
        reader.get_sql("insert_user"),
        {
            "user_name": "SABURO",
            "email": "saburo@test.com",
            "age": 28
        }
    )

    print("===== UPDATE =====")

    runner.update(
        reader.get_sql("update_user"),
        {
            "user_id": 1,
            "user_name": "TARO_UPDATE",
            "email": "taro_new@test.com",
            "age": 22
        }
    )

    print("===== DELETE =====")

    runner.delete(
        reader.get_sql("delete_user"),
        {
            "user_id": 2
        }
    )

    print("===== TRANSACTION TEST =====")

    try:

        runner.begin()

        runner.insert(
            reader.get_sql("insert_user"),
            {
                "user_name": "ROLLBACK_TEST",
                "email": "rollback@test.com",
                "age": 99
            }
        )

        raise Exception("強制エラー")

        runner.commit()

    except Exception:

        print("ROLLBACK実行")

        runner.rollback()

    return "TEST END"

九、架构说明（設計説明）

1 SqlReader

责任：
	•	从 Snowflake Stage 读取 SQL
	•	按 name 解析 SQL
	•	返回 SQL

优点：
	•	SQL 与代码分离
	•	SQL 可版本管理
	•	便于维护

⸻

2 SqlRunner

责任：

统一 SQL 执行

提供：

方法	说明
select	查询
insert	插入
update	更新
delete	删除
batch_insert	批量插入
begin	事务开始
commit	提交
rollback	回滚

4 为什么不需要 DB连接

在 Snowflake 内部执行：

def main(session: Session):

Snowflake 会自动注入 Session

因此：
	•	不需要 JDBC
	•	不需要密码
	•	不需要连接池
