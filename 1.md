4. 共通クラス実装
SqlReader クラス (common/sql_reader.py)
Stage上のファイルをMD5で監視し、変更がある場合のみ再読み込みします。
import hashlib
import re
from snowflake.snowpark import Session
import io

class SqlReader:
    """
    Stage上のSQLファイルを読み込み、キャッシュするクラス。
    MD5ハッシュを用いて、ファイルの更新を検知します。
    """
    _cache = {}  # { 'stage_path': {'md5': str, 'queries': dict} }

    @classmethod
    def get_queries(cls, session: Session, stage_path: str) -> dict:
        """
        StageからSQLファイルを読み込み、パースして返します。
        """
        # 1. Stage上のファイルのMD5を取得 (LISTコマンドを使用)
        # stage_path は '@DB.SCHEMA.STAGE_NAME/file.sql' の形式
        file_info = session.sql(f"LIST {stage_path}").collect()
        if not file_info:
            raise FileNotFoundError(f"SQL file not found in stage: {stage_path}")
        
        # LIST結果の md5 カラム(index 2)を取得
        current_md5 = file_info[0][2]

        # 2. キャッシュチェック
        if stage_path in cls._cache and cls._cache[stage_path]['md5'] == current_md5:
            return cls._cache[stage_path]['queries']

        # 3. ファイルの読み込み (ローカルに保存せずストリームで処理)
        print(f"DEBUG: Reading SQL from stage: {stage_path}")
        stream = session.file.get_stream(stage_path)
        content = io.BytesIO(stream.read()).read().decode('utf-8')
        
        # 4. パース処理 (-- name: key の形式を抽出)
        queries = {}
        # 正規表現で '-- name: 名前' とその後のSQL文を分離
        pattern = re.compile(r'--\s*name:\s*(\w+)\s*(.*?)(?=--\s*name:|$)', re.DOTALL)
        matches = pattern.findall(content)
        
        for name, sql in matches:
            queries[name] = sql.strip()

        # 5. キャッシュの更新
        cls._cache[stage_path] = {
            'md5': current_md5,
            'queries': queries
        }
        
        return queries

SqlRunner クラス (common/sql_runner.py)
実行とトランザクション制御を担当します。
from snowflake.snowpark import Session
from snowflake.snowpark.exceptions import SnowparkSQLException

class SqlRunner:
    """
    SQLの実行、CRUD操作、トランザクション管理を行うクラス。
    """
    def __init__(self, session: Session):
        self.session = session

    def fetch_all(self, sql: str, params: dict = None):
        """検索系SQLを実行し、全件をリスト(Row)で返します。"""
        try:
            return self.session.sql(sql, params).collect()
        except Exception as e:
            raise e

    def execute_update(self, sql: str, params: dict = None) -> int:
        """更新系(CUD)SQLを実行し、影響件数を返します。"""
        try:
            # SnowflakeのSQL実行結果から 'number of rows inserted/updated/deleted' を取得
            result = self.session.sql(sql, params).collect()
            # 通常、更新結果は最初の行の最初のカラムに件数が入る
            return int(result[0][0]) if result else 0
        except Exception as e:
            raise e

    def begin_transaction(self):
        self.session.sql("BEGIN").collect()

    def commit(self):
        self.session.sql("COMMIT").collect()

    def rollback(self):
        self.session.sql("ROLLBACK").collect()

    def execute_batch(self, sql: str, params_list: list) -> int:
        """
        簡易的な一括実行（トランザクション制御付き）
        """
        total_affected = 0
        self.begin_transaction()
        try:
            for params in params_list:
                total_affected += self.execute_update(sql, params)
            self.commit()
            return total_affected
        except Exception as e:
            self.rollback()
            raise e

5. Streamlit テスト画面 (streamlit_app.py)
SnowflakeのStreamlit機能でそのまま動くコードです。
import streamlit as st
from snowflake.snowpark.context import get_active_session
# 同一ディレクトリまたはパスが通っている前提
# from common.sql_reader import SqlReader
# from common.sql_runner import SqlRunner

# Streamlit in Snowflake 上では、ファイル分割されている場合、
# インポート設定が必要ですが、ここでは1つのスクリプト例として記述します。

def main():
    st.set_page_config(layout="wide")
    st.title("❄️ Snowpark CRUD Test UI")
    
    session = get_active_session()
    reader = SqlReader()
    runner = SqlRunner(session)
    
    STAGE_SQL_PATH = "@DEMO_DB.PUBLIC.SQL_STAGE/user_queries.sql"

    try:
        # SQLの読み込み（キャッシュ機能付き）
        queries = reader.get_queries(session, STAGE_SQL_PATH)
        st.sidebar.success("SQL Queries loaded from Stage (Cached)")
    except Exception as e:
        st.error(f"Failed to load SQL: {e}")
        return

    # --- UI Layout ---
    tab1, tab2, tab3 = st.tabs(["Read / Search", "Create / Update", "Transaction Test"])

    with tab1:
        st.header("Search Data")
        if st.button("Fetch All Users"):
            data = runner.fetch_all(queries['select_all'])
            st.dataframe(data, use_container_width=True)
        
        st.divider()
        user_id_input = st.number_input("Search by ID", step=1, value=1)
        if st.button("Search"):
            data = runner.fetch_all(queries['select_by_id'], params={"user_id": user_id_input})
            st.write(data)

    with tab2:
        st.header("CRUD Operations")
        col1, col2 = st.columns(2)
        with col1:
            st.subheader("Insert / Update")
            c_id = st.number_input("User ID", step=1, key="c_id")
            c_name = st.text_input("Name", key="c_name")
            c_email = st.text_input("Email", key="c_email")
            
            if st.button("Insert User"):
                count = runner.execute_update(queries['insert_user'], 
                                             {"user_id": c_id, "user_name": c_name, "email": c_email})
                st.success(f"Inserted {count} rows.")
                
            if st.button("Update User"):
                count = runner.execute_update(queries['update_user'], 
                                             {"user_id": c_id, "user_name": c_name, "email": c_email})
                st.info(f"Updated {count} rows.")

        with col2:
            st.subheader("Delete")
            d_id = st.number_input("User ID to Delete", step=1, key="d_id")
            if st.button("Delete User", type="primary"):
                count = runner.execute_update(queries['delete_user'], {"user_id": d_id})
                st.warning(f"Deleted {count} rows.")

    with tab3:
        st.header("Transaction & Rollback Test")
        st.write("2件のデータを挿入しようとしますが、2件目でエラー(重複ID)を発生させてロールバックを確認します。")
        
        if st.button("Run Failing Transaction"):
            runner.begin_transaction()
            try:
                # 1件目
                runner.execute_update(queries['insert_user'], {"user_id": 99, "user_name": "Success", "email": "ok@ex.com"})
                st.write("Row 1 inserted...")
                # 2件目 (わざとID重複などでエラーにするか、不正なSQLを流す)
                runner.execute_update("INSERT INTO INVALID_TABLE ...") 
                runner.commit()
            except Exception as e:
                runner.rollback()
                st.error(f"Transaction Rolled Back! Error: {e}")
                
        if st.button("Check ID 99 exists?"):
            data = runner.fetch_all(queries['select_by_id'], {"user_id": 99})
            st.write("Result (Should be empty if rolled back):", data)

if __name__ == "__main__":
    main()

6. 実装のポイント解説
 * SqlReader (Stage 参照 & MD5):
   * session.file.get_stream() を使用することで、Snowflakeの計算リソース上のメモリ内でファイルを処理します。ディスクへの一時保存が不要なため、セキュリティと速度に優れています。
   * LIST コマンドで取得できるMD5ハッシュ値を _cache（クラス変数）に保持。同一セッション内での重複読み込みを防止し、ファイルが更新された場合のみパースを再実行します。
 * SqlRunner (影響件数とバインド変数):
   * :user_id のようなバインド変数形式を採用しています。これにより、SQLインジェクションを防ぎつつ、実行計画の再利用が促進されます。
   * collect() の戻り値から「挿入/更新件数」を抽出し、整数で返すように共通化しています。
 * トランザクション管理:
   * BEGIN, COMMIT, ROLLBACK を明示的に呼び出すメソッドを用意しました。Streamlitのテスト画面（Tab 3）で、エラー発生時に正しくロールバックされることを確認できます。
 * SQLファイル構造:
   * -- name: キー名 というコメント行をトリガーにして、一つのファイルに複数のSQLを集約・管理できるようにしています。これによりファイル管理が煩雑になりません。
